class Market
{
	// CI Object
	protected $ci;

	public function __construct()
	{
		$this->ci =& get_instance();
	}
	
	
	/**
	*	Accepts a request
	*	
	*	@param int id		transaction_id
	*/
	public function request_accept( $id )
	{
		// load transaction
		$T = new Transaction();
		$T->where('id', $id)->include_related("recipient", "*", NULL, TRUE)->get();
		
		// change status
		$T->accepted();
		
		// @TODO: send notification
	}
	
	
}
