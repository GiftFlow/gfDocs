class User_openid {

	var $CI;
	
	/**
	 * Constructor
	 */
	function User_openid()
	{
		// Obtain a reference to the ci super object
		$this->CI =& get_instance();
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Get account by account id
	 *
	 * @access public
	 * @param string $account_id
	 * @return object account object
	 */
	function get_by_id($account_id)
	{
		$this->CI->load->model('account_model');
		
		return $this->CI->account_model->get_by_id($account_id);
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Get account by username
	 *
	 * @access public
	 * @param string $username
	 * @return object account object
	 */
	function get_by_username($username)
	{
		$this->CI->load->model('account_model');
		
		return $this->CI->account_model->get_by_username($username);
	}
	
		
	/**
	 * Get account openid
	 *
	 * @access public
	 * @param string $account_id
	 * @return int
	 */
	function get_openid_by_account($account_id)
	{
		$this->CI->load->model('account_openid_model');
		
		return $this->CI->account_openid_model->get_by_account_id($account_id);
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Get account openid
	 *
	 * @access public
	 * @param string $openid
	 * @return int
	 */
	function get_openid($openid)
	{
		$this->CI->load->model('account_openid_model');
		
		return $this->CI->account_openid_model->get_by_openid($openid);
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Add account openid
	 *
	 * @access public
	 * @param string $openid
	 * @param int $account_id
	 * @return void
	 */
	function add_openid($openid, $account_id)
	{
		$this->CI->load->model('account_openid_model');
		$this->CI->account_openid_model->insert($openid, $account_id);
	}
	
	
	// --------------------------------------------------------------------
	
	/**
	 * Create account
	 *
	 * @access public
	 * @param string $username
	 * @param string $password
	 * @return int insert id
	 */
	function create($username, $email = NULL, $password = NULL)
	{
		// Create password hash using phpass
		if ($password !== NULL)
		{
			$this->CI->load->plugin('account/phpass');
			$hasher = new PasswordHash(PHPASS_HASH_STRENGTH, PHPASS_HASH_PORTABLE);
			$hashed_password = $hasher->HashPassword($password);
		}
		
		$this->CI->load->model('account/account_model');
		$account_insert_id = $this->CI->account_model->create($username, $email, isset($hashed_password) ? $hashed_password : NULL);
		
		return $account_insert_id;
	}
	// --------------------------------------------------------------------
	
	/**
	 * Delete account openid
	 *
	 * @access public
	 * @param string $openid
	 * @return void
	 */
	function delete_openid($openid)
	{
		$this->CI->load->model('account_openid_model');
		$this->CI->account_openid_model->delete($openid);
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Add details to account
	 *
	 * @access public
	 * @param int $account_id
	 * @param array $attributes
	 * @return void
	 */
	function update_details($account_id, $attributes)
	{
		$this->CI->load->model('account_model');
		$this->CI->account_model->googleupdate($account_id, $attributes);
	}
	
	
	// --------------------------------------------------------------------
	
	/**
	 * Retrieve all available countries
	 *
	 * @access public
	 * @return object
	 */
	function get_all_countries()
	{
		$this->CI->load->model('ref_country_model');
		
		return $this->CI->ref_country_model->get_all();
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Retrieve all available languages
	 *
	 * @access public
	 * @return object
	 */
	function get_all_languages()
	{
		$this->CI->load->model('ref_language_model');
		
		return $this->CI->ref_language_model->get_all();
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Retrieve all available zoneinfos
	 *
	 * @access public
	 * @return object
	 */
	function get_all_zoneinfos()
	{
		$this->CI->load->model('ref_zoneinfo_model');
		
		return $this->CI->ref_zoneinfo_model->get_all();
	}
	
}
