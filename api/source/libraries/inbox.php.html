	{
		$current_user_id = $this->CI->session->userdata('user_id');
		$query = $this->CI->db ->from("message_deliveries AS md ")
						->join("messages AS m ", "md.message_id = m.id")
						->join("users AS u ", "m.author_id = u.id")
						->join("transactions AS t", "m.transaction_id = t.id")
						->where("t.id", $id)
