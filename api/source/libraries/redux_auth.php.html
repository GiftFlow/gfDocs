/**
* Redux Authentication 2
*/
class redux_auth
{
	/**
	 * CodeIgniter global
	 *
	 * @var string
	 **/
	protected $ci;

	/**
	 * account status ('not_activated', etc ...)
	 *
	 * @var string
	 **/
	protected $status;

	/**
	 * __construct
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function __construct()
	{
		$this->ci =& get_instance();
		$email = $this->ci->config->item('email');
		$this->ci->load->library('email', $email);
		$this->ci->load->model('user');
	}
	
	/**
	 * Activate user.
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function activate($code)
	{
		return $this->ci->user->activate($code);
	}
	
	/**
	 * Deactivate user.
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function deactivate($email)
	{
	    return $this->ci->user->deactivate($email);
	}
	
	/**
	 * Change password.
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function change_password($identity, $old, $new)
	{
        return $this->ci->user->change_password($identity, $old, $new);
	}

	/**
	 * forgotten password feature
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function forgotten_password( $email )
	{
		$forgotten_password = $this->ci->user->forgotten_password_key($email);
		
		
		// Get user information.
		$profile = $this->ci->user->profile($email);

		$message = $this->ci->load->view($this->ci->config->item('email_templates').'forgotten_password', $data, true);
			
		$this->ci->email->clear();
		$this->ci->email->set_newline("\r\n");
		$this->ci->email->from('', '');
		$this->ci->email->to($profile->email);
		$this->ci->email->subject('Email Verification (Forgotten Password)');
		$this->ci->email->message($message);
		
		// Return true if email successfully sent
		if($this->ci->email->send())
			return true;
		else
			return false;
	}
	
	/**
	 * undocumented function
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function forgotten_password_complete($code)
	{
	    	$identity    	= $this->ci->config->item('identity');
	    	$profile          	= $this->ci->user->profile($code);
		$forgotten_password_complete = $this->ci->user->forgotten_password_complete($code);

		if ($forgotten_password_complete)
		{
			$data = array('identity'    => $profile->{$identity},
				         'new_password' => $this->ci->user->new_password);
            
			$message = $this->ci->load->view($this->ci->config->item('email_templates').'new_password', $data, true);
				
			$this->ci->email->clear();
			$this->ci->email->set_newline("\r\n");
			$this->ci->email->from('', '');
			$this->ci->email->to($profile->email);
			$this->ci->email->subject('New Password');
			$this->ci->email->message($message);
			return $this->ci->email->send();
		}
		else
			return false;
	}

	/**
	 * register
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function register($email, $password, $name)
	{
	    	$email_activation = $this->ci->config->item('email_activation');
	    	$email_folder     = $this->ci->config->item('email_templates');
		
		// Skip email validation?
		if (!$email_activation)
		{
			return $this->ci->user->register($email, $password, $name);
		}
		// Email validation routine
		else
		{
			// create db entry
			$register = $this->ci->user->register($email, $password, $name);
            
			if (!$register) 
				return false;
			
			// deactive new user
			$deactivate = $this->ci->user->deactivate($email);

			if (!$deactivate)
				return false;

			$activation_code = $this->ci->user->activation_code;

			$data = array(  'password'   	=> $password,
        					'email'      	=> $email,
        					'name'		=> $name,
        					'activation' 	=> $activation_code);
            
            		// load email template, activation. fill with data array.
			$message = $this->ci->load->view($email_folder.'activation', $data, true);
            		
			$this->ci->email->clear();
			$this->ci->email->set_newline("\r\n");
			$this->ci->email->from($this->ci->config->item('email_from_address'), $this->ci->config->item('email_from_name'));
			$this->ci->email->to($email);
			$this->ci->email->subject($this->ci->config->item('email_subject'));
			$this->ci->email->message($message);
			
			$status = $this->ci->email->send();
			echo $this->ci->email->print_debugger();
			return $status;
		}
	}
	
	/**
	 * login
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function login($identity, $password)
	{
		return $this->ci->user->login($identity, $password);
	}
	
	/**
	 * logout
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function logout()
	{
	    $identity = $this->ci->config->item('identity');
	    $this->ci->session->unset_userdata($identity);
		$this->ci->session->sess_destroy();
	}
	
	/**
	 * logged_in
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function logged_in()
	{
	    	$identity = $this->ci->config->item('identity');
		return ($this->ci->session->userdata($identity)) ? true : false;
	}
	
	/**
	 * Profile
	 *
	 * @return void
	 * @author Mathew
	 **/
	public function profile()
	{
	    $session  = $this->ci->config->item('identity');
	    $identity = $this->ci->session->userdata($session);
	    return $this->ci->user->profile($identity);
	}
	
}
