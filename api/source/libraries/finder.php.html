class Finder
{

	protected $ci;
	
	// Typical search parameters
	var $keyword;
	var $location;
	
	// Location details
	var $address;
	var $latitude;
	var $longitude;
	
	// Search radius
	var $radius = 10000;
	
	// LatLng bounds defined using search radius
	var $bounds;
	
	// Type of search
	var $type = 'gift';
	
	// Tag ID
	var $tag_id;
	
	// Stores subquery strings which will be merged in a union
	var $subquery;

	// Result
	var $result;
	
	// Return data format
	var $format;
	
	public function __construct()
	{
		$this->ci =& get_instance();
	}
	
	public function locate_ip( $ip = NULL )
	{
		if(empty($ip)) $ip = $this->ci->input->ip_address();
		$url = "http://ipinfodb.com/ip_query.php?ip=".$ip."&timezone=false";
		$xml = simplexml_load_file( $url );
		if(!empty($xml->City)&&!empty($xml->RegionName))
		{
			$data['city'] = $xml->City;
			$data['state'] = $xml->RegionName;
			$data['latitude'] = $xml->Latitude;
			$data['longitude'] = $xml->Longitude;
			return $data;
		}
		else return false;
	}
	
	protected function _set_bounds()
	{
		if(!isset($this->latitude)||!isset($this->longitude)) return false;
		
		// How wide is one degree?
		$latitude_degree = 69;
		$longitude_degree = ( ( 3.14 * cos(deg2rad($this->latitude)) * 6367449 ) / (180 * 1000 ) ) / 1.609;
		
		// How wide is the search radius (in degrees)?
		$latitude_radius = $this->radius  / $latitude_degree;
		$longitude_radius = $this->radius / $latitude_degree;
		$this->bounds = array ( "latitude"=>array ( 	"upper"=> ( $this->latitude + $latitude_radius ),
										"lower" => ( $this->latitude - $latitude_radius ) ),
						"longitude"=>array(	"upper"=> ( $this->longitude + $longitude_radius ),
										"lower" => ( $this->longitude - $longitude_radius ) ) );		
		$this->ci->db->where("L.latitude BETWEEN ".$this->bounds['latitude']['lower']." AND ".$this->bounds['latitude']['upper']);
		$this->ci->db->where("L.longitude BETWEEN ".$this->bounds['longitude']['lower']." AND ".$this->bounds['longitude']['upper']);
		$this->ci->db->select("( 3959 * acos( cos( radians( ".$this->latitude." ) ) * cos( radians( L.latitude ) ) * cos( radians( L.longitude ) - radians(".$this->longitude.") ) + sin( radians(".$this->latitude.") ) * sin( radians( L.latitude ) ) ) ) AS distance");

	}
	public function popular_tags()
	{
		$tags = array();
		$this->_set_bounds();
		$this->ci->db	->select('t.name AS tag, COUNT(g.id) AS count')
					->from('goods_tags AS gt ')
					->join('goods AS g ', 'g.id=gt.good_id')
					->join('tags AS t ', 't.id = gt.tag_id')
					->join('locations AS L ','L.id=g.location_id')
					->where('g.winning_transaction_id',0)
					->where('g.type','gift')
					->group_by('t.name')
					->order_by('count', 'desc');
		$this->ci->db->select('t.id AS tag_id');
		$query = $this->ci->db->get();

		return $query->result();
	}
	
	
	public function tag_search ( $tag = NULL )
	{
		if(!empty($tag))
		{
			// Find the tag_id
			$tag_query = $this->ci->db->select("id")->like('name', $tag)->get('tags');
			if($tag_query->num_rows==0) return false;
			$tag = $tag_query->row();
			$this->tag_id = $tag->id;
		}
		$this->query();
		$this->format_result();
		return $this->result;
	}
	
	
	public function search()
	{
		$this->query();
		
		if( $this->format == "JSON" )
		{
			$this->_format_result();
			if(!empty($this->latitude)&&!empty($this->longitude)&&!empty($this->address))
				$data['center'] = (object) array ( "latitude"=>$this->latitude, "longitude"=> $this->longitude, "address"=>$this->address);
			$data['results'] = $this->result;
			return json_encode( $data );
		}
		else
		{
			$this->_format_result();
			return $this->result;
		}
	}
	
	public function query()
	{
		// Geocode if needed
		if(!empty($this->location)&&(empty($this->latitude)||empty($this->longitude)))
		{
			$L = new Location();
			if(!$L->geocode( $this->location )) return false;
			$this->latitude = $L->latitude;
			$this->longitude = $L->longitude;
			$this->address = $L->address;
		}

		// Find any matching tags
		$this->_set_tag( $this->keyword );
		
		// Search by keyword
		$this->subquery[] = $this->_title_query();

		// If keyword matches a tag, filter by it
		if ( !empty($this->tag_id) )
			$this->subquery[] = $this->_tag_query();

		if(count($this->subquery)==2)
		{
			
			$union = "(".$this->subquery[0]." ) UNION ( ".$this->subquery[1].")";
			$query = $this->ci->db->query($union);
		}
		elseif(count($this->subquery)==1)
			$query = $this->ci->db->query($this->subquery[0]);
			
		// Store result
		$this->result = $query->result();
	}
	
	
	private function _basic_query()
	{
		// Determine geographic boundaries
		$this->_set_bounds();
		// Find goods
		$this->ci->db	->select('G.id, G.title, G.id, L.latitude, L.longitude, U.screen_name AS author')
						->from('goods AS G ')
						->join("locations AS L ", "L.id = G.location_id")
						->join('users AS U ', 'G.user_id = U.id')
						->where('G.type', $this->type);
		
		// If bounds are set, then add where clauses
		if ( !empty($this->bounds))
		{
			$this->ci->db->where("L.latitude BETWEEN ".$this->bounds['latitude']['lower']." AND ".$this->bounds['latitude']['upper']);
			$this->ci->db->where("L.longitude BETWEEN ".$this->bounds['longitude']['lower']." AND ".$this->bounds['longitude']['upper']);		
		}
	}
	
	
	private function _tag_query()
	{
		$this->ci->db->_reset_select();
		$this->_basic_query();
		$this->ci->db->join("goods_tags AS GT ", "GT.good_id=G.id")
			->join("tags AS T ", "GT.tag_id=T.id")
			->where('T.id = '.$this->tag_id);
		$subquery = $this->ci->db->_compile_select();
		$this->ci->db->_reset_select();
		return $subquery;
	}
	
	
	private function _title_query()
	{
		$this->ci->db->_reset_select();
		$this->_basic_query();
		$this->ci->db->like("G.title", $this->keyword);
		$subquery = $this->ci->db->_compile_select();
		$this->ci->db->_reset_select();
		
		return $subquery;
	}
	
	private function _format_result()
	{
		foreach($this->result as $key=>$val)
		{
			$val->html = result_row($val->id, array("author", "location"));
		}
	}
	
	private function _set_tag( $tag )
	{
		// Find the tag_id
		$tag_query = $this->ci->db->select("id")->like('name', $tag)->get('tags');
		if($tag_query->num_rows==0) return false;
		$tag = $tag_query->row();
		$this->tag_id = $tag->id;
	}

}
