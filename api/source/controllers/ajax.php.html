class Ajax extends Controller {

	var $data;
	var $G;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
		$this->load->library('finder');
	}
	
	/**
	*	Encodes a string into a usable location
	*	@param string $address
	*	@param boolean $JSON
	*/
	public function geocode( $address = NULL, $JSON = FALSE )
	{
		if(empty($address)) $address = $_POST['where'];
		$L = new Location();
		if(!$L->geocode( $address )) return false;
		$coordinates = (object) array ( "latitude"=>$L->latitude, "longitude"=> $L->longitude, "address"=>$L->address);
		
		if($JSON)
		{
			echo json_encode($coordinates);
		}
		else 
		{
			return $coordinates;
		}
	}
	
	
	/**
	*	Get search results of any type good
	*	Used extensively in the find section. Usually the client makes AJAX calls to one of the
	*	methods that extend this one, such as gift_search(). The search results are generated
	*	using Finder::geosearch().
	*
	* 	@param string $good_type
	*	@return JSON
	*/
	public function search( $good_type = 'gift' )
	{
		$this->finder->search_keyword = $_POST['what'];
		$this->finder->location_keyword = $_POST['where'];
		if(!empty($_POST['address']))
			$this->finder->address = $_POST['address'];
		$this->finder->latitude = $_POST['latitude'];
		$this->finder->longitude = $_POST['longitude'];
		$this->finder->format = "JSON";
		$this->finder->limit(50);
		$this->finder->where_type( $good_type );
		echo $this->finder->geosearch();
	}
	
	/**
	*	Get search results, limited to gifts
	*
	*	@return JSON
	*/
	public function gift_search()
	{
		$this->search( 'gift' );
	}
	
	/**
	*	Get search results, limited to needs
	*
	*	@return JSON
	*/
	public function need_search()
	{
		$this->search( 'need' );
	}
	

	/**
	*	Get a GiftFlow based on provided source
	*
	*	The source variable should be passed along as part of either the POST or GET data.
	*	this variable will directly determine which method the Finder library uses to get the 
	*	search results. Example values of the source variable are followers (default), nearby_gifts
	*	and latest_gifts.
	*
	*	@return JSON
	*/
	public function giftflow()
	{
		$this->load->library('finder');
		!empty($_REQUEST['source']) ? $source = $_REQUEST['source'] : $source = 'followers';
		!empty($_REQUEST['offset']) ? $offset = $_REQUEST['offset'] : $offset = 0;
		!empty($_REQUEST['limit']) ? $limit = $_REQUEST['limit'] : $limit = 10;
		echo $this->finder->limit($limit, $offset)->giftflow();
	}
	
	public function tags()
	{
		$term = $_POST['term'];
		$this->db  ->select('Concat(t.name," (",COUNT(g.id), ")") AS label, t.name AS value ', FALSE)
					->from('goods_tags AS gt ')
					->join('goods AS g ', 'g.id=gt.good_id')
					->join('tags AS t ', 't.id = gt.tag_id')
					//->where('g.winning_transaction_id',0)
					->where('g.type','gift')
					->like('t.name',$term)
					->group_by('t.name')
					->order_by('COUNT(g.id)', 'desc');
		//$this->db->select('t.id AS tag_id');
		$search = $this->db->get();
		echo json_encode($search->result_array());
	}
	public function upload_photo()
	{
		// set config details
		$config['upload_path'] = './uploads/';
		$config['allowed_types'] = 'gif|jpg|png';
		$config['max_size']	= '500';
		$config['max_width']  = '1024';
		$config['max_height']  = '768';
		$config['file_name'] = uniqid();
		
		// load library
		$this->load->library('upload', $config);
	
		if ( ! $this->upload->do_upload())
		{
			echo $this->upload->display_errors();
		}	
		else
		{
			$this->data['upload'] = $this->upload->data();
			$G = new Good($_POST['id']);
			$P = new Photo();
			$P->data = $this->data['upload'];
			$P->add();
			$G->save_photo($P);
			$G->save_default_photo($P);
			echo "
    • ";
		}
	}
	
	public function delete_photo()
	{
		$P = new Photo($_POST['id']);
		$P->delete();
		echo json_encode(array("result"=>"true", "id"=>$_POST['id']));
	}
}
