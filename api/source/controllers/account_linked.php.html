/*
 * Account_linked Controller
 */
class Account_linked extends Controller {
	
	/**
	 * Constructor
	 */
	function Account_linked()
	{
		parent::Controller();
		
		// Load the necessary stuff...
		$this->load->config('account');
		$this->load->helper(array('ssl', 'date'));
		$this->load->language('sign_in');
		$this->load->language('account_linked');
		$this->load->library(array('form_validation', 'auth'));
		$this->load->library('user_openid');
		$this->U = new User($this->session->userdata('user_id'));
		$this->data = $this->util->parse_globals();
		$this->data['U'] = $this->U;
	}
	
	/**
	 * Linked accounts
	 */
	function index()
	{
		
		
		$this->data['title'] = "Manage Linked Accounts";
		// Enable SSL?
		maintain_ssl($this->config->item("ssl_enabled"));
		
		// Redirect unauthenticated users to signin page
		if ( ! $this->auth->is_signed_in()) 
		{
			$this->session->set_userdata('sign_in_redirect', 'account_linked/index');
			redirect('member/login');
		}
		
		// Retrieve sign in user
		$data['account'] = $this->user_openid->get_by_id($this->session->userdata('user_id'));
		
		// Delete a linked account
		if ( $this->input->post('openid'))
		{
			if ($this->input->post('openid'))
			{
				$this->user_openid->delete_openid($this->input->post('openid'));
			}
			$this->session->set_flashdata('linked_info', $this->lang->line('linked_linked_account_deleted'));
			redirect('account_linked/index');
		}
		
		// Check for linked accounts
		$data['num_of_linked_accounts'] = 0;
		
		// Get OpenID accounts
		if ($data['openid_links'] = $this->user_openid->get_openid_by_account($this->session->userdata('user_id'))) 
		{
			foreach ($data['openid_links'] as $index => $openid_link) 
			{
				if (strpos($openid_link->openid, 'google.com')) $data['openid_links'][$index]->provider = 'google';				
				else $data['openid_links'][$index]->provider = 'openid';
				
				$data['num_of_linked_accounts']++;
			}
		}
		
		$this->data['active_link'] = 'links';

		if( !empty( $this->U->facebook_id ) )
		{
			$this->data['links']['facebook']['enabled'] = TRUE;
			$this->data['links']['facebook']['id'] = $this->U->facebook_id;
		}
		else
			$this->data['links']['facebook']['enabled'] = FALSE;
			

		$this->load->view('header', $this->data);
		$this->load->view('account/menu', $this->data);
		$this->load->view('account/links', $this->data);
		$this->load->view('footer', $this->data);
	}
	
}
