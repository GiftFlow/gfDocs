/*
 * Sign_in_openid Controller
 */
class Sign_in_openid extends Controller {

	/**
	 * Constructor
	 */
	function Sign_in_openid()
	{
		parent::Controller();
		
		// Load the necessary stuff...
		$this->load->config('account');
		$this->load->helper('ssl');
		$this->load->plugin('openid');
		$this->load->language('sign_in');
		$this->load->language('account_linked');
		$this->load->library('authentication');
	}
	
	function index()
	{
		// Enable SSL?
		maintain_ssl($this->config->item("ssl_enabled"));
		
		$this->load->library('form_validation');
	
		// Setup form validation
		$this->form_validation->set_error_delimiters('', '');
		$this->form_validation->set_rules(array(
			array('field'=>'sign_in_openid_url', 'label'=>'lang:sign_in_openid_url', 'rules'=>'trim|required')
		));
		
		// Run form validation
		if ($this->form_validation->run()) 
		{
			// Get OpenID store object
			$store = new Auth_OpenID_FileStore($this->config->item("openid_file_store_path"));
			
			// Get OpenID consumer object
			$consumer = new Auth_OpenID_Consumer($store);
		
			// Begin OpenID authentication process
			if ( ! $auth_request = $consumer->begin($this->input->post('sign_in_openid_url'))) 
			{
				$data['sign_in_openid_error'] = sprintf($this->lang->line('sign_in_invalid_openid'), $this->lang->line('sign_in_openid'));
			}
			else
			{
				// Create sreg_request (Simple Registration)
				if ($sreg_request = Auth_OpenID_SRegRequest::build(array('nickname', 'email', 'fullname', 'gender', 'dob', 'postcode', 'country', 'language', 'timezone'))) 
					$auth_request->addExtension($sreg_request);
				
				// Built redirect URL
				$redirect_url = $auth_request->redirectURL(base_url(), site_url('account/sign_in_openid/verify'));
				
				// Page (or possibly popup) redirect
				header("Location: ".$redirect_url);
			}
		}
		
		$this->load->view('sign_in_openid', isset($data) ? $data : NULL);
	}
		
	/**
	 * Verify openid's authentication response
	 *
	 * @access public
	 * @return void
	 */
	function verify()
	{
		// Enable SSL?
		maintain_ssl($this->config->item("ssl_enabled"));
		
		// Get OpenID store object
		$store = new Auth_OpenID_FileStore($this->config->item("openid_file_store_path"));
		
		// Get OpenID consumer object
		$consumer = new Auth_OpenID_Consumer($store);
		
		// Complete authentication process using server response
		$response = $consumer->complete(site_url('account/sign_in_openid/verify'));
		
		// Check the response status
		if ($response->status == Auth_OpenID_SUCCESS) 
		{
			$this->load->library('user');
		
			// Extract claimed identifier
			$openid = $response->getDisplayIdentifier();
			
			// Check if openid is already link to a user
			if ($user_openid = $this->user->get_openid($openid))
			{
				// Link OpenID account
				if ($this->authentication->is_signed_in())
				{
					// Check if linked user is current user
					$user_openid->account_id == $this->session->userdata('account_id') ?
						$this->session->set_flashdata('linked_error', sprintf($this->lang->line('linked_linked_with_this_account'), $this->lang->line('sign_in_openid'))) :
							$this->session->set_flashdata('linked_error', sprintf($this->lang->line('linked_linked_with_another_account'), $this->lang->line('sign_in_openid')));
							
					redirect('account/account_linked');
				}
				// Sign in with OpenID
				else
				{
					// Run sign in routine
					$this->authentication->sign_in($user_openid->account_id);
				}
			}
			else
			{
				// Link OpenID account
				if ($this->authentication->is_signed_in())
				{
					// Add OpenID to user				
					$this->user->add_openid($openid, $this->session->userdata('account_id'));
					$this->session->set_flashdata('linked_info', sprintf($this->lang->line('linked_linked_with_your_account'), $this->lang->line('sign_in_google')));
					
					redirect('account/account_linked');
				}
				// Sign in with OpenID
				else
				{
					$openid_openid['openid'] = $openid;
					
					// Extract Simple Registration data
					if ($sreg_resp = Auth_OpenID_SRegResponse::fromSuccessResponse($response))
					{
						$sreg = $sreg_resp->contents();
						if (isset($sreg['nickname'])) $openid_openid['username'] = $sreg['nickname'];
						if (isset($sreg['email'])) $openid_openid['email'] = $sreg['email'];
						if (isset($sreg['fullname'])) $openid_openid['fullname'] = $sreg['fullname'];
						if (isset($sreg['gender'])) $openid_openid['gender'] = $sreg['gender'];
						if (isset($sreg['dob'])) $openid_openid['dateofbirth'] = $sreg['dob'];
						if (isset($sreg['postcode'])) $openid_openid['postalcode'] = $sreg['postcode'];
						if (isset($sreg['country'])) $openid_openid['country'] = $sreg['country'];
						if (isset($sreg['language'])) $openid_openid['language'] = $sreg['language'];
						if (isset($sreg['timezone'])) $openid_openid['timezone'] = $sreg['timezone'];
					}
					
					$this->session->set_userdata('openid_openid', $openid_openid);
					
					redirect('account/sign_in_openid/complete');
				}
			}
		}
		elseif ($response->status == Auth_OpenID_CANCEL) 
		{
			$this->session->set_flashdata('sign_in_openid_error', sprintf($this->lang->line('sign_in_cancelled'), $this->lang->line('sign_in_openid')));
		} 
		elseif ($response->status == Auth_OpenID_FAILURE) 
		{
			$this->session->set_flashdata('sign_in_openid_error', sprintf($this->lang->line('sign_in_failed'), $this->lang->line('sign_in_openid')));
		}
		redirect('account/sign_in_openid');
	}
	
	/**
	 * Complete openid's authentication process
	 *
	 * @access public
	 * @return void
	 */
	function complete()
	{
		// Enable SSL?
		maintain_ssl($this->config->item("ssl_enabled"));
		
		// Retrieve 'openid_openid' from session
		$openid_openid = $this->session->userdata('openid_openid');
		
		// If 'openid_openid' doesn't exist, redirect to home
		if ( ! $openid_openid) redirect('');
		
		$this->load->library(array('form_validation', 'user'));
		
		// Setup form validation
		$this->form_validation->set_error_delimiters('', '');
		$this->form_validation->set_rules(array(
			array('field'=>'sign_in_complete_username', 'label'=>'lang:sign_in_username', 'rules'=>'trim|required|alpha_dash|min_length[2]|max_length[16]')
		));
		
		// Run form validation
		if ($this->form_validation->run()) 
		{
			// Check if user name is taken
			if ($this->username_check($this->input->post('sign_in_complete_username')) === TRUE)
			{
				$data['sign_in_complete_username_error'] = $this->lang->line('sign_in_complete_username_taken');
			}
			else
			{
				// Create user
				$user_id = $this->user->create($this->input->post('sign_in_complete_username'), isset($openid_openid['email']) ? $openid_openid['email'] : NULL);

				// Add OpenID to user				
				$this->user->add_openid($openid_openid['openid'], $user_id);
				
				// Add details to user
				$attributes = array();
				if (isset($openid_openid['fullname'])) $attributes['fullname'] = $openid_openid['fullname'];
				if (isset($openid_openid['dateofbirth'])) $attributes['dateofbirth'] = $openid_openid['dateofbirth'];
				if (isset($openid_openid['gender'])) $attributes['gender'] = $openid_openid['gender'];
				if (isset($openid_openid['postalcode'])) $attributes['postalcode'] = $openid_openid['postalcode'];
				if (isset($openid_openid['country'])) $attributes['country'] = $openid_openid['country'];
				if (isset($openid_openid['language'])) $attributes['language'] = $openid_openid['language'];
				if (isset($openid_openid['timezone'])) $attributes['timezone'] = $openid_openid['timezone'];
				$attributes = $this->user->validate_details($attributes);
				$this->user->update_details($user_id, $attributes);
				
				// Unset openid data
				$this->session->unset_userdata('openid_openid');
				
				// Run sign in routine
				$this->authentication->sign_in($user_id);
			}
		}
		
		$data['provider'] = 'openid';
		
		$this->load->view('sign_in_complete', $data);
	}
	
	/**
	 * Check if a username exist
	 *
	 * @access public
	 * @param string
	 * @return bool
	 */
	function username_check($username)
	{
		return $this->user->get_by_username($username) ? TRUE : FALSE;
	}
	
}
