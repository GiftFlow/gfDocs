class Root extends Controller {

	var $data;
	
	function Root()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
	}
	/**
	 * index
	 *
	 **/
	function index()
	{
		if(!$this->auth->validate(1))
			$this->_signup();
		else
			$this->_dash();
	}
	function _signup()
	{
		$this->data['title'] = "Welcome";
		$this->load->view('header', $this->data);
		$this->load->view('index');
		$this->load->view('footer', $this->data);
	}
	function _dash()
	{
		$this->benchmark->mark('dash_start');
		$this->data['title'] = "Your Dashboard";
		$this->load->library('finder');
		$this->data['googlemaps'] = TRUE;
		
		$U = new User($this->data['logged_in_user_id']);
		$this->finder->format = 'object';
		
		$this->data['giftflow'] = $this->finder->giftflow();
		
		$this->data['your_gifts'] = $U->gifts();
		$this->data['your_needs'] = $U->needs();
		$this->data['your_requests'] = $U->requests;
		
		$this->load->view('header', $this->data);
		$this->load->view('member/index', $this->data);
		$this->load->view('footer', $this->data);
		$this->benchmark->mark('dash_end');
	}

	function lost()
	{
		$this->data['title'] = "Oops! Page cannot be found.";

		$this->load->view('header', $this->data);
		$this->load->view('404', $this->data);
		$this->load->view('footer', $this->data);
	}
	function restricted()
	{
		$this->data['title'] = "Restricted area";

		$this->load->view('header', $this->data);
		$this->load->view('restricted', $this->data);
		$this->load->view('footer', $this->data);
	}
	
	function test()
	{
		$this->load->library('finder');
		$this->finder->follower_search();
	}
}
