class Welcome extends Controller {

	function Welcome()
	{
		parent::Controller();	
		$this->output->enable_profiler(TRUE);
	}
	
	function index()
	{
		$this->load->view('welcome_message');
	}
	function mapper()
	{
		$m = new Location();
		$m->address = "6780 Bethesda Arno Rd.";
		$m->city = "Thompsons Station";
		$m->state = "TN";
		$m->user_id = 10;
		$m->save();
		$u = new User(10);
		$u->default_location_id = $m->id;
		$u->save();
		echo $u->default_location_id;
	}
	function geocode()
	{
		
		$this->benchmark->mark('maps_start');
		$this->load->model('location');
		$this->benchmark->mark('maps_end');
		$this->benchmark->elapsed_time('maps_start', 'maps_end');
		//echo $this->maps->geocode("6780 Bethesda Arno Rd Thompons Station TN");
		//echo $this->maps->geocode("493 College St. New Haven CT");
		$this->benchmark->mark('distance_start');
		for($i=0; $i<1000; $i++)
		{
			$this->location->distance("35.7747489","-86.7738427","41.3111214","-72.9257758","m");
		}
		$this->benchmark->mark('distance_end');
		$this->benchmark->elapsed_time('distance_start', 'distance_end');
		
		$this->benchmark->mark('comparison_start');
		$one = new Location();
		$one->latitude = "35.7747489";
		$one->longitude = "-86.7738427";
		
		for($i=0; $i<1000; $i++)
		{
			$two = new Location();
			$two->latitude = "41.3111214";
			$two->longitude = "-72.9257758";
			$two->compare($one);
			unset($two);
		}
		
		unset($one);
		$this->benchmark->mark('comparison_end');
		$this->benchmark->elapsed_time('comparison_start', 'comparison_end');
		
		$this->benchmark->mark('mixed_start');
		$one = new Location();
		$one->latitude = "35.7747489";
		$one->longitude = "-86.7738427";
		
		for($i=0; $i<1000; $i++)
		{
			$this->location->distance($one->latitude, $one->longitude,"41.3111214","-72.9257758","m");
		}
		
		
		$this->benchmark->mark('mixed_end');
		$this->benchmark->elapsed_time('mixed_start', 'mixed_end');
	}
	//
	// 1mil: 66,890.974 / s
	// 2mil: 66,270.589 / s
	// 500k:  67,346.416 / s
}
