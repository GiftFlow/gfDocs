class Mail extends Controller {

	var $data;
	
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
	}


	function index()
	{
		$this->inbox();
	}
	function compose()
	{
		if(isset($_POST['submit']))
		{ 
			$m = new Message();
			$m->subject = $this->input->post('subject');
			$m->body =  $this->input->post('body');
			$m->recipients =  explode("&", $this->input->post('recipients'));
			$m->author_id =  16; //$this->session->userdata('user_id');
			//$m->thread_id =  9;//$this->input->post('thread');
			$m->compose();
			if(count($m->error->all) > 0)
			{
				echo $m->error->string;
			}
			else
			{	
				echo "success";
			}
		}
		else
		{
			$this->load->view("mail/compose");
		}
	}
	function inbox()
	{
		//$user_id = $this->uri->segment(3);
		$user_id = $this->session->userdata('user_id');
		$u = new User($user_id);
		$u->message_delivery->get();
		$this->data['inbox'] = $u->message_delivery->all;
		$this->data['title'] = "Inbox";
		$this->load->view('header', $this->data);
		$this->load->view('mail/inbox',$this->data);
		$this->load->view('footer', $this->data);
	}
	function message( $id )
	{
		$t = new Thread( $id );
		$t->messages->get();
		$this->data['thread'] = $t;
		$this->data['messages'] = $t->messages->all;
		$this->data['title'] = $t->subject;
		
		$this->load->view('header', $this->data);
		$this->load->view('mail/message',$this->data);
		$this->load->view('footer', $this->data);
	}
