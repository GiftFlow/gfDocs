	function login( $redirect = FALSE )
	{
		if(!empty($_POST))
		{
			$this->u = $this->auth->login();
			if(count($this->u->error->all) > 0)
			{
				$this->session->set_flashdata('error', $this->u->error->string);//$this->data['alert_error'] = $this->u->error->all;
				redirect('login'); //$this->_login_form();
			}
			else
			{
				if($this->input->post('redirect'))
				{
					$q = $this->db->where('id', $this->input->post('redirect'))->get('redirects',1);
					$r = $q->row();
					redirect($r->url);
					
				}
				else
				{
					redirect('');
				}
			}
		}
		else
		{
			$this->data['redirect'] = $redirect;
			$this->_login_form();
		}
	}
	
	function logout()
	{
		$this->auth->logout();
		redirect('');
	}
	
	function register()
	{
		$this->load->library('recaptcha');

		if(!empty($_POST))
		{
			if (!$this->recaptcha->check_answer($this->input->ip_address(),$this->input->post('recaptcha_challenge_field'),$this->input->post('recaptcha_response_field')))
			{
				$this->session->set_flashdata('error', "You did not correctly input the words in the image. Please try again.");
				redirect('register');
			}
			$this->u = $this->auth->register();
			if(count($this->u->error->all) > 0)
			{
				$this->session->set_flashdata('error', $this->u->error->all);
				redirect('register');
			}
			else
			{
				$this->_register_success();
			}
		}
		else
		{
			$this->_register_form();
		}
	}
	
	function profile()
	{
		$this->benchmark->mark('profile_start');
		
		if ( !$user_id = $this->uri->segment(2) )
		{
			if(empty($this->data['userdata']['user_id']))
			{
				redirect('lost');
			}
			else
			{
				$user_id = $this->session->userdata('user_id');
			}
		}
		
		$u = new User();
		$u	->where('id',$user_id)
			->include_related('default_location', '*', NULL, TRUE)
			->include_related('default_photo', '*', NULL, TRUE)
			->get();
		$u->gift = $u->gifts();
		$u->need = $u->needs();
		$u->stats();
		$this->data['following'] = $u->following();
		if(!empty($this->data['logged_in_user_id']))
			$this->data['is_following'] = $u->is_followed_by($this->data['logged_in_user_id']);
		$this->data['u'] = $u;
		$this->data['completed_gifts'] = $u->completed_gifts();
		$this->data['completed_requests'] = $u->completed_requests();
		$this->data['title'] = $u->screen_name." | Profile";
		$this->load->view('header', $this->data);
		$this->load->view('member/profile', $this->data);
		$this->load->view('footer', $this->data);
		
		$this->benchmark->mark('profile_end');
		$this->benchmark->elapsed_time('profile_start','profile_end');
	}

	function change_password()
	{	    
	    $this->form_validation->set_rules('old', 'Old password', 'required');
	    $this->form_validation->set_rules('new', 'New Password', 'required|matches[new_repeat]');
	    $this->form_validation->set_rules('new_repeat', 'Repeat New Password', 'required');
	    $this->form_validation->set_error_delimiters('
', '
');
	    
	    if ($this->form_validation->run() == false)
	    {
	        $data['content'] = $this->load->view('redux_auth/change_password', null, true);
	        $this->load->view('redux_auth/template', $data);
	    }
	    else
	    {
	        $old = $this->input->post('old');
	        $new = $this->input->post('new');
	        
	        $identity = $this->session->userdata($this->config->item('identity'));
	        
	        $change = $this->redux_auth->change_password($identity, $old, $new);
		
    		if ($change)
    		{
    			$this->logout();
    		}
    		else
    		{
    			echo "Password Change Failed";
    		}
	    }
	}

	function follow( $id )
	{
		$U = new User($this->data['logged_in_user_id']);
		$F = new User( $id );
		$U->save_following($F);
		$this->session->set_flashdata('success', 'Now following '.$F->screen_name.'.');
		redirect('profile/'.$id);
	}
	
	function unfollow( $id )
	{
		$U = new User($this->data['logged_in_user_id']);
		$F = new User( $id );
		$U->delete_following($F);
		$this->session->set_flashdata('success', 'No longer following '.$F->screen_name.'.');
		redirect('profile/'.$id);
	}

	function facebook( $key1 = null, $val1 = null, $key2 = null, $val2 = null )
	{
		// Handles the initial facebook connect request. Redirects user to Facebook's OAuth page. The data will be returned to /assets/facebook.php.
		if(empty($key1))
		{
			$config = array (	"client_id"=>111637438874755,
								"scope" => "offline_access,user_location,user_photos,email,publish_stream",
								"redirect_uri"=>"http://www.giftflow.org/assets/facebook.php" );
			
			redirect("https://graph.facebook.com/oauth/authorize?".http_build_query($config));
		}
		// Once the user has been authorized, this code parses the authorization code and sends the necessary data to the Auth class.
		else
		{
			$access = $key1.'='.$val1;
			if(!empty($key2)&&!empty($val2)) $access .= $key2.'='.$val2;
			$facebook_data = json_decode(file_get_contents("https://graph.facebook.com/me?".$access));
			if($key1=="access_token") $facebook_data->token = $val1;
			$this->auth->facebook( $facebook_data );
		}	
	}
