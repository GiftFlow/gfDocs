      function index()
      {
            // Enable SSL?
            maintain_ssl($this->config->item("ssl_enabled"));
            
            // Get OpenID store object
            $store = new Auth_OpenID_FileStore($this->config->item("openid_file_store_path"));
            
            // Get OpenID consumer object
            $consumer = new Auth_OpenID_Consumer($store);
            
            // Begin OpenID authentication process
            
            // If there is an error, where should we send them?
            if (!$auth_request = $consumer->begin($this->config->item("openid_google_discovery_endpoint")))
            {
                  // If logged in, to the Linked accounts screen
                  if ($this->session->userdata('user_id'))
                  {
                        $this->session->set_flashdata('linked_error', sprintf($this->lang->line('sign_in_invalid_discovery'), $this->lang->line('sign_in_google')));
                        redirect('account/links');
                  }
                  // Else, to the login screen
                  else
                  {
                        $this->session->set_flashdata('sign_in_error', sprintf($this->lang->line('sign_in_invalid_discovery'), $this->lang->line('sign_in_google')));
                        redirect('login');
                  }
            }
            
            // There isn't an error! Onwards!
            
            // Create ax request (Attribute Exchange)
            $ax_request = new Auth_OpenID_AX_FetchRequest;
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/namePerson/friendly', 1, TRUE, 'username'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/contact/email', 1, TRUE, 'email'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/namePerson', 1, TRUE, 'fullname'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/birthDate', 1, TRUE, 'dateofbirth'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/person/gender', 1, TRUE, 'gender'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/contact/postalCode/home', 1, TRUE, 'postalcode'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/contact/country/home', 1, TRUE, 'country'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/pref/language', 1, TRUE, 'language'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/pref/timezone', 1, TRUE, 'timezone'));
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/namePerson/first', 1, TRUE, 'firstname')); // google only
            $ax_request->add(Auth_OpenID_AX_AttrInfo::make('http://axschema.org/namePerson/last', 1, TRUE, 'lastname')); // google only
            $auth_request->addExtension($ax_request);
            
            // Built redirect URL
            $redirect_url = $auth_request->redirectURL(base_url(), site_url('sign_in_google/verify'));
            $redirect_url .= "&openid.ns.ext2=http://specs.openid.net/extensions/oauth/1.0&openid.ext2.consumer=giftflow.org&openid.ext2.scope=".urlencode('http://www.google.com/m8/feeds/');

            // Page (or possibly popup) redirect
            header("Location: " . $redirect_url);
      }

            // Get OpenID consumer object
            $consumer = new Auth_OpenID_Consumer($store);
            
            // Complete authentication process using server response
            $response = $consumer->complete(site_url('sign_in_google/verify'));
            
            // Check the response status
            
            // If success....
            if ($response->status == Auth_OpenID_SUCCESS)
            {
                  $this->load->library('user_openid');
                  
                  // Extract claimed identifier
                  $openid = $response->getDisplayIdentifier();
                  
                  // Is this OpenID already linked to a user?
                  if ($user_openid = $this->user_openid->get_openid($openid))
                  {
                        // Is this user already logged in? 
                        if ($this->session->userdata('user_id'))
                        {
                              // User both logged in and the OpenID matches the user's account. Link OpenID to this account!
                              if ($user_openid->account_id == $this->session->userdata('user_id'))
                              {
                                    $this->session->set_flashdata('success', sprintf($this->lang->line('linked_linked_with_this_account'), $this->lang->line('sign_in_google')));
                              }
                              
                              // User logged in but the OpenID belongs to someone else. Throw an error and redirect them back to the login page.
                              else
                              {
                                    $this->session->set_flashdata('error', sprintf($this->lang->line('linked_linked_with_another_account'), $this->lang->line('sign_in_google')));
                              }
                        }
                        // User isn't signed in, begin login routine
                        else
                        {
                              $this->auth->login($user_openid->account_id);
                        }
                  }
                  
                  // OpenID isn't linked to GiftFlow. Begin linking routine.
                  else
                  {
                        $openid_google['openid'] = $openid;
                        
                        if ($ax_args = Auth_OpenID_AX_FetchResponse::fromSuccessResponse($response))
                        {
                              $ax_args = $ax_args->data;
                              if (isset($ax_args['http://axschema.org/namePerson/friendly'][0]))

      /**
       * Complete google's authentication process
       *
       * @access public
       * @return void
       */
      function complete()
      {
            // Enable SSL?
            maintain_ssl($this->config->item("ssl_enabled"));
            
            // Retrieve 'openid_google' from session
            $openid_google = $this->session->userdata('openid_google');
            
            // If 'openid_google' doesn't exist, redirect to home
            if (!$openid_google)
                  redirect('');
            
            $this->load->library(array(
                  'form_validation',
                  'user_openid'
            ));
            
            // Create user
            $user_id = $this->session->userdata('user_id');
            
            // Add OpenID to user				
            $this->user_openid->add_openid($openid_google['openid'], $user_id);
            
            // Add details to user
            $attributes = array();
            if (isset($openid_google['fullname']))
                  $attributes['fullname'] = $openid_google['fullname'];
            if (isset($openid_google['dateofbirth']))
                  $attributes['dateofbirth'] = $openid_google['dateofbirth'];
            if (isset($openid_google['gender']))
                  $attributes['gender'] = $openid_google['gender'];
            if (isset($openid_google['postalcode']))
                  $attributes['postalcode'] = $openid_google['postalcode'];
            if (isset($openid_google['country']))
                  $attributes['country'] = $openid_google['country'];
            if (isset($openid_google['language']))
                  $attributes['language'] = $openid_google['language'];
            if (isset($openid_google['timezone']))
                  $attributes['timezone'] = $openid_google['timezone'];
            if (isset($openid_google['firstname']))
                  $attributes['firstname'] = $openid_google['firstname']; // google specific
            if (isset($openid_google['lastname']))
                  $attributes['lastname'] = $openid_google['lastname']; // google specific
            
            $this->user_openid->update_details($user_id, $attributes);
            
            // Unset openid data
            $this->session->unset_userdata('openid_google');
            
            // Run sign in routine
            $this->auth->login($user_id);
            
            
            $data['provider'] = 'google';
            
            $this->data['title'] = "Your Settings";
            $this->load->view('header', $this->data);
            $this->load->view('openid_auth/contact_detail', $this->data);
            $this->load->view('footer', $this->data);
      }
      
      function send_signed_request($http_method, $url, $auth_header=null,
                             $postData=null, $returnResponseHeaders=true) {
  $curl = curl_init($url);
  curl_setopt($curl, CURLOPT_RETURNTRANSFER, true);
  curl_setopt($curl, CURLOPT_FAILONERROR, false);
  curl_setopt($curl, CURLOPT_SSL_VERIFYPEER, false);

  if ($returnResponseHeaders) {
    curl_setopt($curl, CURLOPT_HEADER, true);
  }

  switch($http_method) {
    case 'GET':
      if ($auth_header) {
        curl_setopt($curl, CURLOPT_HTTPHEADER, array($auth_header));
      }
      break;
    case 'POST':
      $headers = array('Content-Type: application/atom+xml', $auth_header);
      curl_setopt($curl, CURLOPT_HTTPHEADER, $headers);
      curl_setopt($curl, CURLOPT_POST, 1);
      curl_setopt($curl, CURLOPT_POSTFIELDS, $postData);
      break;
    case 'PUT':
      $headers = array('Content-Type: application/atom+xml', $auth_header);
      curl_setopt($curl, CURLOPT_HTTPHEADER, $headers);
      curl_setopt($curl, CURLOPT_CUSTOMREQUEST, $http_method);
      curl_setopt($curl, CURLOPT_POSTFIELDS, $postData);
      break;
    case 'DELETE':
      $headers = array($auth_header);
      curl_setopt($curl, CURLOPT_HTTPHEADER, $headers);
      curl_setopt($curl, CURLOPT_CUSTOMREQUEST, $http_method);
      break;
  }
  $response = curl_exec($curl);
  if (!$response) {
    $response = curl_error($curl);
  }
  curl_close($curl);
  return $response;
}
