class Give extends Controller {

	var $data;
	var $G;
	var $U;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
		$this->hooks =& load_class('Hooks');

	}

	public function index()
	{
		$this->data['title'] = "Give";
		$this->data['active'] = '';
		$this->data['pending'] = '';
		$this->data['past'] = '';
		$this->data['css'][] = "datatables.css";
		$this->data['js'][] = 'in-field-label.js';
		$this->data['js'][] = "jquery-datatables.php";
		if(!empty($this->data['logged_in_user_id']))
		{
			$this->U = new User($this->data['logged_in_user_id']);
			$this->data['goods'] = $this->U->gifts();
		}
		else
		{
			$this->data['goods'] = NULL;
		}
		$this->load->view('header', $this->data);
		$this->load->view('give/index', $this->data);
		$this->load->view('footer');
	}
	public function add()
	{
		$this->auth->bouncer(1);

		$this->U = new User($this->data['logged_in_user_id']);
		$item = "gift";
		if(!empty($_POST))
		{
			// Create location object
			$L = new Location();
			$L->raw = $this->input->post('location');
			$L->user_id = $this->data['logged_in_user_id'];
			$L->validate();
			if(!empty($L->duplicate_id))
				$L = new Location($L->duplicate_id);
			else
				$L->save();

			$this->G = new Good();
			$this->G->title = $this->input->post("title");
			$this->G->type = $item;
			if ( $this->G->save( array( $L, $this->U) ) )
			{
				// Save location object to user and gift
				$this->U->save_location($L);
				$this->U->default_location->get();
				if(!$this->U->default_location->exists()) $this->U->save_default_location($L);
				//$this->G->save($L);
				
				// Save tags
				$tags = explode(",", $_POST['tags']);
				foreach ( $tags as $tag )
					$this->G->add_tag( trim($tag) );
				
				// Hook: 'gift_new'
				$this->hooks->call('gift_new', $this);
				
				$this->session->set_flashdata('success','Gift saved! Add more information below if you like.');
				$this->data['alert_success'][] = 'Gift saved! Add more information below if you like.';
				redirect('gift/'.$this->G->id);
				/*
				//DO UPLOAD
				$config['upload_path'] = './assets/uploads/';
				$config['allowed_types'] = 'gif|jpg|png';
				$config['max_size']	= '100';
				$config['max_width']  = '1024';
				$config['max_height']  = '768';
				
				$this->load->library('upload', $config);
			
				if ( ! $this->upload->do_upload())
				{
					$this->data['alert_error'] = $this->upload->display_errors();
					$this->_add_form($item);
				}
				*/
			}
			else
			{
				$this->data['alert_error'] = $this->G->error->all;
				$this->_add_form($item);
			}
			
		}
		else
		{
			redirect('give');
		}
	}
	function history()
	{
		$this->auth->bouncer(1);
		$this->data['css'][] = "datatables.css";
		$this->data['js'][] = "jquery.php";
		$this->data['js'][] = "jquery-datatables.php";
		$this->data['title'] = "Gift History";
		$U = new User($this->data['logged_in_user_id']);
		$U->gift->order_by('created', 'DESC')->get();
		$this->data['gifts'] = $U->gift->all;
		$this->load->view('header', $this->data );
		$this->load->view('give/history', $this->data);
		$this->load->view('footer', $this->data );
	}
	
	function view()
	{
		// see gift controller
	}
	protected function _add_form( $item )
	{
		if(empty($this->G)) $this->G = new Good();
		// Load the htmlform extension, so we can generate the form.
		$this->G->load_extension('htmlform');
		
		// This is how are form will be rendered
		$U = new User($this->data['logged_in_user_id']);
		
		$U->location->get();
		foreach($U->location->all as $val)
		{
			$this->data['locations'][$val->id] = $val->title;
		}
		$this->data['type'] = $item;
		$this->data['current_location'] = NULL;
		$this->data['title'] = "Add a good";

		$this->load->view('header', $this->data );
		$this->load->view('market/add/'.$item, $this->data);
		$this->load->view('footer', $this->data );
	}
	
}
