/**
*	Community central will help link people to other people.
*
*	@author Brandon Jackson
*	@package Controllers
*/
class Community extends Controller {

	var $data;
	var $G;
	var $U;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
		$this->load->library('finder');
		
		
		// Set user's location
		if( !empty($this->session->userdata['location_address']))
		{
			$location['address'] = $this->session->userdata['location_address'];
			if(!empty($this->session->userdata['location_street_address']))
				$location['street_address'] = $this->session->userdata['location_street_address'];
			if(!empty($this->session->userdata['location_city']))
				$location['city'] = $this->session->userdata['location_city'];
			if(!empty($this->session->userdata['location_state']))
				$location['state'] = $this->session->userdata['location_state'];
			if(!empty($this->session->userdata['location_latitude']))
				$location['latitude'] = $this->session->userdata['location_latitude'];
			if(!empty($this->session->userdata['location_longitude']))
				$location['longitude'] = $this->session->userdata['location_longitude'];
			$this->data['location'] = $location;
		}
		elseif( !empty($this->data['location']->city)&&!empty($this->data['location']->state))
		{
			// Finds location based on user's IP address. Passes back city, state, latitute and longitude (or returns false).
			$this->data['location'] = $this->finder->locate_ip( $this->input->ip_address() );
			$this->data['location']['address'] = $this->data['location']['city'].', '.$this->data['location']['state'];
		}
		if(!empty($this->data['logged_in_user_id'])) $this->U = new User($this->data['logged_in_user_id']);
	}

	public function index()
	{
		redirect('community/facebook');
	}	

	public function facebook()
	{
		if(!empty($this->U->facebook_id))
		{
			$this->load->library('facebook');
			
			// Load list of friends from Facebook API
			$this->data['friends'] = $this->facebook->friends( $this->data['logged_in_user_id'] );
			foreach($this->data['friends'] as $key=>$val)
			{ 
				$this->data['friends'][$key]->following = $val->is_followed_by($this->data['logged_in_user_id']);
			}
		}

		// Page Title
		$this->data['title'] = "Find Your Friends on Facebook | Community";
		
		// Includes the needed Google Maps references in the header
		$this->data['googlemaps'] = TRUE;
		
		$this->data['active_link'] = 'facebook';
		
		$this->load->view('header', $this->data);
		$this->load->view('community/menu', $this->data);
		$this->load->view('community/facebook', $this->data);
		$this->load->view('footer', $this->data);
	}
	
	public function gmail()
	{
		// Page Title
		$this->data['title'] = "Find Your Friends on Gmail | Community";
		
		// Includes the needed Google Maps references in the header
		$this->data['googlemaps'] = TRUE;
		
		$this->data['active_link'] = 'gmail';
		
		$this->load->view('header', $this->data);
		$this->load->view('community/menu', $this->data);
		$this->load->view('community/gmail', $this->data);
		$this->load->view('footer', $this->data);
	}
	

}
