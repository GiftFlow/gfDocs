application/controllers/ajax.php


class Ajax extends Controller {

	var $data;
	var $g;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
		$this->load->library('finder');
	}

	public function geocode( $address = NULL, $json = FALSE )
	{
		if(empty($address)) $address = $_POST['where'];
		$L = new Location();
		if(!$L->geocode( $address )) return false;
		$coordinates = (object) array ( "latitude"=>$L->latitude, "longitude"=> $L->longitude, "address"=>$L->address);
		if($json) echo json_encode($coordinates);
		else return $coordinates;
	}
	
	public function gift_search()
	{
		$this->finder->keyword = $_POST['what'];
		$this->finder->location = $_POST['where'];
		if(!empty($_POST['address']))
			$this->finder->address = $_POST['address'];
		$this->finder->latitude = $_POST['latitude'];
		$this->finder->longitude = $_POST['longitude'];
		$this->finder->format = "JSON";
		echo $this->finder->search();
	}
	
	public function tags()
	{
		$term = $_POST['term'];
		$this->db  ->select('Concat(t.name," (",COUNT(g.id), ")") AS label, t.name AS value ', FALSE)
					->from('goods_tags AS gt ')
					->join('goods AS g ', 'g.id=gt.good_id')
					->join('tags AS t ', 't.id = gt.tag_id')
					//->where('g.winning_transaction_id',0)
					->where('g.type','gift')
					->like('t.name',$term)
					->group_by('t.name')
					->order_by('COUNT(g.id)', 'desc');
		//$this->db->select('t.id AS tag_id');
		$search = $this->db->get();
		echo json_encode($search->result_array());
	}
	public function upload_photo()
	{
		// set config details
		$config['upload_path'] = './uploads/';
		$config['allowed_types'] = 'gif|jpg|png';
		$config['max_size']	= '500';
		$config['max_width']  = '1024';
		$config['max_height']  = '768';
		$config['file_name'] = uniqid();
		
		// load library
		$this->load->library('upload', $config);
	
		if ( ! $this->upload->do_upload())
		{
			echo $this->upload->display_errors();
		}	
		else
		{
			$this->data['upload'] = $this->upload->data();
			$G = new Good($_POST['id']);
			$P = new Photo();
			$P->data = $this->data['upload'];
			$P->add();
			$G->save_photo($P);
			$G->save_default_photo($P);
			echo "
    • ";
		}
	}
	
	public function delete_photo()
	{
		$P = new Photo($_POST['id']);
		$P->delete();
		echo json_encode(array("result"=>"true", "id"=>$_POST['id']));
	}
	
}
