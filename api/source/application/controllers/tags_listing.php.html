class Tags_listing extends Controller {

	var $data;
	var $g;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
	}

	public function index()
	{
		
	}
	public function view()
	{
		$tag = $this->uri->segment(2);
		$T = new Tag();
		$T->where('name', $tag)->get();
		$T->gift->get();
		$this->data['gifts'] = $T->gift->all;
		$this->data['title'] = $tag;
		$this->data['tag'] = $tag;
		$this->load->view('header', $this->data );
		$this->load->view('tag/view', $this->data);
		$this->load->view('footer', $this->data );
	}
	
}
