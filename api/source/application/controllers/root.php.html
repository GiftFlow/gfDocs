class Root extends Controller {

	var $data;
	
	function Root()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
	}
	/**
	 * index
	 *
	 **/
	function index()
	{
		if(!$this->auth->validate(1))
			$this->_signup();
		else
			$this->_dash();
	}
	function _signup()
	{
		$this->data['title'] = "Welcome";
		$this->load->view('header', $this->data);
		$this->load->view('index');
		$this->load->view('footer', $this->data);
	}
	function _dash()
	{
		$this->data['title'] = "Your Dashboard";
		$U = new User($this->data['userdata']['user_id']);
		$U->following->get();
		if(count($U->following->all)>0)
		{
			foreach($U->following->all as $val) $following[] = $val->id;
			$G = new Good();
			$G->where_in_related_user('id', $following)->get();
			$this->data['follow_feed'] = $G->all;
		}
		else
		{
			$this->data['follow_feed'] = false;
		}
		$this->load->library('facebook');
		if(!empty($U->facebook_id)) $this->data['friends'] = $this->facebook->friends( $U->id );
		$this->data['your_gifts'] = $U->gifts();
		$this->data['your_needs'] = $U->needs();
		$this->data['your_requests'] = $U->requests();
		$this->load->view('header', $this->data);
		$this->load->view('member/index', $this->data);
		$this->load->view('footer', $this->data);
	}

	function lost()
	{
		$this->data['title'] = "Oops! Page cannot be found.";

		$this->load->view('header', $this->data);
		$this->load->view('404', $this->data);
		$this->load->view('footer', $this->data);
	}
	function restricted()
	{
		$this->data['title'] = "Restricted area";

		$this->load->view('header', $this->data);
		$this->load->view('restricted', $this->data);
		$this->load->view('footer', $this->data);
	}
}
