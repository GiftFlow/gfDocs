	function __construct()
	{
		parent::Controller();
		
		// Load external classes
		$this->load->helper('elements');
		$this->load->library('market');
		
		// Set some class-wide variables
		$this->good_id = $this->uri->segment(2);
		$this->method = $this->uri->segment(3);
		$this->param = $this->uri->segment(4);
		
		// Initialize Good
		if(!empty($this->good_id))
		{
			$this->G = new Good ( $this->good_id );
			$this->G->user->get();
			$this->G->location->get();
			$this->G->photo->get();
			$this->G->default_photo->get();
			$this->G->tag->get();
			$this->G->request = $this->G->requests();
		}
		
		// Parse global data
		$this->util->config();
		$this->data = $this->util->parse_globals();
		
		if(!empty($this->G))
			$this->data['G'] = $this->G;
		$this->data['content'] = array();


	}

	function view()
	{
		// Begin Routing
		$go = "_".$this->method;
		
		// If this auto-routed function exists, execute it
		if( method_exists( __CLASS__, $go ) )
		{
			$this->$go();
		}
		
		// If not, determine the proper index page to display
		else
		{
			// route gift author to giver dashboard
			if($this->_restrict(FALSE))
			{
				$this->_giver_dashboard();
			}
			// Route everyone else
			else
			{
				// Logged in...
				if(!empty($this->data['logged_in_user_id']) )
				{
					$this->recipient_status = $this->G->request_status( $this->data['logged_in_user_id'] );
					if( $this->recipient_status['status'] == 'pending' || $this->recipient_status['status'] == "accepted" || $this->recipient_status['status'] == "exchanged" )
						$this->_recipient_dashboard();
					else
						$this->_view();
				}
				// If not logged in...
				else
					$this->_view();
			}
		}
	}

	function _view()
	{
		// Prepare data
		if($this->data['logged_in'])
		{	
			// Load the user's request
			// returns array( 'status'=>{String}, 'request'=>{Transaction object} )
			$your_request = $this->G->request_status( $this->data['logged_in_user_id'] );
			if( $your_request['status'] == "pending" || $your_request['status'] == "accepted" || $your_request['status'] == "ignored" )
			{
				$this->data['requested'] = TRUE;
				$this->data['content']['requests'] = array("title"=>"info","content"=>$this->load->view('gift/includes/requests', array('request'=>$your_request['request'], "type"=>"view"), true));
			}
			else
			{
				$this->data['requested'] = FALSE;
			}
		}
		
		// Prepare output
		
		// Load main gift module
		$this->data['right'] = $this->load->view('gift/includes/view', $this->data, true);
		
		// Title
		$this->data['title'] = $this->G->title." | A Gift from ".$this->G->user->screen_name;
		
		// Breadcrumbs
		$this->data['breadcrumbs'][] = array("title"=>"Gifts", "href"=>site_url("gift"));
		$this->data['breadcrumbs'][] = array ("title"=>$this->G->title);
		
		// Load views
		$this->load->view('header', $this->data);
		$this->load->view('gift/view', $this->data);
		$this->load->view("footer", $this->data);
	}

	function _request()
	{
		// Create Transaction
		$T = new Transaction();
		
		// Create User
		$U = new User( $this->data['logged_in_user_id'] );
		
		// Set request reason (if set)
		if(!empty($_POST['reason'])) $T->reason = $_POST['reason'];
		
		// Save transaction & relationships
		$T->save();
		$T->save_recipient($U);
		$T->save_gift($this->G);
				
		// @TODO: Send notification
				
		// Set flashdata & redirect
		$this->session->set_flashdata('success', 'Gift requested!');
		redirect('gift/'. $this->good_id );
	}
	
	function _unrequest()
	{
		// Create Transaction
		$T = new Transaction();
		
		// Load Request
		$T	->where_related_gift( 'id',  $this->good_id )
			->where_related_recipient( 'id', $this->data['logged_in_user_id'] )
			->get();
			
		// Delete Request
		//$this->G->delete_request($T);
		//$U = new User( $this->data['logged_in_user_id'] );
		//$U->delete($T);
		$T->delete();
		
		// Set flashdata & redirect
		$this->session->set_flashdata('success', 'Request canceled.');
		redirect('gift/'. $this->good_id );
	}

	function _giver_dashboard_save()
	{
		$U = new User($this->data['logged_in_user_id']);

		$this->G->title = $_POST["title"];
		$this->G->description = $_POST["description"];
		$this->G->save();
		
		/*
		* Save locaiton
		*/
		
		//Create location object
		$L = new Location();
		$L->raw = $this->input->post('location');
		$L->user_id = $this->data['logged_in_user_id'];
		
		//Validate it to make sure this location doesn't already exist
		$L->validate();
		
		// If it does already exist, load existing location
		if(!empty($L->duplicate_id))
			$L = new Location($L->duplicate_id);
		
		// If it doesn't already exist, save new lcoation
		else
			$L->save();
		
		// Save location object to user and gift
		$U->save($L);
		$this->G->save($L);
		
		/*
		* Save tags
		*/
		
		// Tags are passed as a comma delimited list. Explode this list into an array.
		$tags = explode(",", $_POST['tags']);
		
		// Load and delete existing tags
		$T = $this->G->tag->get();
		$this->G->delete($T->all);
		
		// Add new tags (and reload existing tags)
		foreach ( $tags as $tag )
			$this->G->add_tag( $tag );
		
		$U->save_good($this->G);
		$this->session->set_flashdata('success','Changes saved successfully.');
		
		redirect('gift/'.$this->G->id);

	}
	function _abort()
	{
		$T = new Transaction( $this->param );
		$T->recipient->get();
		$T->status = "pending";
		$T->save();
		
		$this->session->set_flashdata("success", $T->recipient->screen_name." is no longer selected as the recipient.");
		redirect('gift/'.$this->good_id);
	}
	function _accept()
	{
		$this->_restrict();
		
		// make sure that request id was passed in the uri
		$request_id = $this->param;
		if( !$request_id  && !$this->data['is_ajax'] ) redirect('gift/'.$this->good_id);
		
		// load transaction
		$T = new Transaction();

		$T->where('id', $request_id )->include_related("recipient", "*", NULL, TRUE)->get();
		
		// change status
		$T->accepted();
		
		// @TODO: send notification

		// set flashdata, redirect
		$this->session->set_flashdata('success', $T->recipient->screen_name."'s request accepted.");
		redirect('gift/'. $this->good_id );
	}
	
	function _delete()

		$this->session->set_flashdata('success', $this->G->title." was deleted successfully."); 
		$this->G->delete();

	function _exchanged()
	{
		// make sure that request id was passed in the uri
		$request_id = $this->param;
		if( !$request_id && !$this->data['is_ajax'] ) redirect('gift/'.$this->good_id);
		
		// load transaction
		$T = new Transaction();
		$T->where( 'id', $request_id )->include_related("recipient", "*", NULL, TRUE)->get();

		if($this->_restrict(FALSE))
			$T->exchanged("giver");
		else
			$T->exchanged("recipient");
		
		// @TODO: send notification

		// set flashdata, redirect
		$this->session->set_flashdata('success', "Gift marked as exchanged.");
		redirect('gift/'. $this->good_id );
	}
	
	function _ignore()
	{
		// make sure that request id was passed in the uri as segment 4
		if( !$this->param && !$this->data['is_ajax'] ) redirect('gift/'.$this->good_id);

		// Load transaction
		$T = new Transaction();
		$T->where('id', $this->param)->include_related("recipient", "*", NULL, TRUE)->get();
		
		// Change status
		$T->ignored();
		
		if( $this->data['is_ajax'] )
		{
			if( count( $this->G->requests( "pending" ) ) > 0 )
			{
				echo "Request ignored.";
				exit;
			}
			else
			{
				echo "Request ignored. There are no more requests at this time.";
				exit;
			}
		}
		else
		{
			// set flashdata, redirect
			$this->session->set_flashdata('success', $T->recipient->screen_name."'s request accepted.");
			redirect('gift/'. $this->good_id );
		}
	}
	function _message()
	{
		$M = new Message();
		$M->body = $_POST['message'];
		$M->author_id = $this->data['logged_in_user_id'];
		$this->G->active_transaction = $this->G->active_transaction();
		$M->transaction_id = $this->G->active_transaction->id;
		$M->recipients = array($this->data['logged_in_user_id'],  $this->G->active_transaction->recipient->id);
		$M->transaction_dialogue();
		redirect( 'gift/' . $this->G->id );
	}
	
	/*
	*	Tests to see if the current logged in user is also the author of the current
	*	good. If yes, then it returns TRUE. If not, then it can do one of two things. 
	*
	*	If $redirect is unspecified or set to TRUE, the user is redirect to the default
	*	view gift page. If $redirect is set to FALSE, the function returns FALSE.
	*	
	*	@param bool redirect
	*	@returns bool OR null + redirect (see above for details)
	*/
	function _restrict( $redirect = TRUE )
	{
		if(!empty($this->data['logged_in_user_id']) && $this->G->user->id==$this->data['logged_in_user_id'] )
			return true;
		else
		{
			if($redirect)
				redirect('gift/'.$this->good_id);
			else
				return false;
		}
	}
