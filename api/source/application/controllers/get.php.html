class Get extends Controller {

	var $data;
	var $G;
	var $N;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
		$this->load->helper('elements');
		$this->hooks =& load_class('Hooks');
	}

	public function index()
	{
		$this->data['title'] = "Get";
		$this->data['active'] = '';
		$this->data['pending'] = '';
		$this->data['past'] = '';
		$this->data['css'][] = "datatables.css";
		$this->data['js'][] = 'in-field-label.js';
		$this->data['js'][] = "jquery-datatables.php";
		if(!empty($this->data['logged_in_user_id']))
		{
			$u = new User($this->data['logged_in_user_id']);
			$u->good->order_by('created', 'DESC')->get();
			$this->data['goods'] = $u->needs();
		}
		else
		{
			$this->data['goods'] = NULL;
		}
		$this->load->view('header', $this->data);
		$this->load->view('get/index', $this->data);	
		$this->load->view('footer', $this->data);
	}
	
	public function add()
	{
		$U = new User($this->data['logged_in_user_id']);
		$item = "need";
		$this->auth->bouncer(1);
		if(!empty($_POST))
		{
			// Create location object
			$L = new Location();
			$L->raw = $this->input->post('location');
			$L->user_id = $this->data['logged_in_user_id'];
			$L->validate();
			if(!empty($L->duplicate_id))
				$L = new Location($L->duplicate_id);
			else
				$L->save();

			$this->G = new Good();
			$this->G->title = $this->input->post("title");
			$this->G->type = $item;
			if(!empty($_POST['description'])) $this->G->description = $this->input->post('description');
			if ( $this->G->save( array( $L, $U ) ) )
			{
				
				// Save location object to user and gift
				$U->save_location($L);
				$U->default_location->get();
				if(!$U->default_location->exists()) $U->save_default_location($L);
				
				$tags = explode(",", $_POST['tags']);
				foreach ( $tags as $tag )
					$this->G->add_tag( trim($tag) );
				
				// Hook: 'need_new'
				$this->hooks->call('need_new', $this);

				// Set flashdata
				$this->session->set_flashdata('success','Need saved! Add more information below if you like.');
				redirect('need/'.$this->G->id.'/edit');
			}
			else
			{
				$this->data['alert_error'] = $this->G->error->all;
				$this->_add_form($item);
			}
		}
		else
		{
			$this->_add_form( $item );
		}
	}
	

	
	
	public function view( $id = NULL )
	{	
		// Store URI segments
		$good_id = $this->uri->segment(2);
		$method = $this->uri->segment(3);
		$param = $this->uri->segment(4);
		
		// Prepare data
		$this->N = new Good ( $good_id );
		$this->N->location->get();
		$this->N->user->get();
		$this->N->default_photo->get();
		$this->N->photo->get();
		$this->N->tag->get();
		
		
		$this->data['N'] = $this->N;

		if(!empty($this->data['logged_in_user_id'])&&$this->N->user->id==$this->data['logged_in_user_id'])
		{
			if($method=='delete')
				$this->_delete_need();
			elseif($method=="edit")
				$this->_edit_need();
			else
				$this->_need_dashboard();
		}
		else
		{
			$this->_view_need();
		}
	}
















	function _view_need()
	{
		$this->data['content'] = array();
		if($this->data['logged_in'])
		{
			$your_requests = $this->G->requests($this->data['logged_in_user_id']);
			if(count($your_requests)>0)
			{
				$requested = TRUE;
				$this->data['content']['requests'] = array('title'=>'requests', 'content'=>$this->load->view('gift/includes/requests', array('request'=>$your_requests, "type"=>"view"), true));
			}
		}
		
		if(!empty($this->G->description))
		{
			$this->data['content']['info'] = array('title'=>'info', 'content'=>$G->description);
		}
		if(count($this->G->tag->all)>0)
		{
			$this->data['content']['tags'] = array('title'=>'tags', 'content'=>$this->load->view('gift/includes/tags', array('tags'=>$this->G->tag->all), true));
		}
		
		// Prep Buttons
		$button_data = array('page'=>"view_gift","G"=>$this->G);
		if(!empty($requested)&&$requested) $button_data['requested'] = TRUE;
		else $button_data['requested'] = FALSE;
		$this->data['buttons'] = $this->load->view('gift/includes/buttons', $button_data, true);
		
		// Other data
		$this->data['title'] = "View Need";
		$this->data['G'] = $this->G;
		
		// Load views
		$this->load->view('header', $this->data);
		$this->load->view('gift/view', $this->data);
		$this->load->view("footer", $this->data);	
	}

	function _delete_need()
	{
		if(!empty($this->data['logged_in_user_id'])&&$this->N->user->id==$this->data['logged_in_user_id'])
		{
			$this->N->delete();
			$this->session->set_flashdata('success', 'Need deleted.');
			redirect('get');
		}
		else
		{
			$this->session->set_flashdata('error', 'Permission denied.');
			redirect('get');
		}
	}
	function _edit_need()
	{
		if(!empty($_POST))
		{
			$U = new User($this->data['logged_in_user_id']);

			$this->N->title = $_POST["title"];
			$this->N->description = $_POST["description"];
			$this->N->save();
			
			// Create location object
			$L = new Location();
			$L->raw = $this->input->post('location');
			$L->user_id = $this->data['logged_in_user_id'];
			$L->validate();
			if(!empty($L->duplicate_id))
				$L = new Location($L->duplicate_id);
			else
				$L->save();
			
			// Save location object to user and gift
			$U->save($L);
			$this->N->save($L);
			
			// Save tags
			$tags = explode(",", $_POST['tags']);
			foreach ( $tags as $tag )
				$this->N->add_tag( trim($tag) );
			
			$U->save_gift($this->N);
			$this->session->set_flashdata('success','Changes saved successfully.');
			redirect('need/'.$this->N->id."/edit");
		}
		else
		{
			$U = new User($this->data['logged_in_user_id']);
			$U->location->get();
			$this->data['G'] = $this->N;
			$this->data['U'] = $U;
			$this->data['backlink'] = TRUE;
			$this->data['content']['photos'] = array('title'=>'photos', 'content'=>$this->load->view('gift/includes/edit/photos', array('data'=>$this->data), true));
			$this->data['content']['info'] = array('title'=>'info', 'content'=>$this->load->view('gift/includes/edit/info', array("data"=>$this->data), true));
			$this->data['content']['tags'] = array('title'=>'tags', 'content'=>$this->load->view('gift/includes/edit/tags', array("data"=>$this->data), true));
			$this->data['content']['location'] = array('title'=>'location', 'content'=>$this->load->view('gift/includes/edit/location', array("data"=>$this->data), true));
			$this->data['content']['save'] = array('title'=>'save', 'content'=>$this->load->view('gift/includes/edit/save', array("data"=>$this->data), true));
			
			// Prep Buttons
			$button_data = array("G"=>$this->N);
			$this->data['buttons'] = $this->load->view('gift/includes/buttons/need_edit', $button_data, true);
			
			$this->data['title'] = "Edit Gift";
			$this->load->view('header', $this->data);
			$this->load->view('gift/edit', $this->data);
			$this->load->view("footer", $this->data);		
		}
	}
	function _need_dashboard()
	{
		$this->data['content'] = array();

		if(!empty($this->N->description))
		{
			$this->data['content']['info'] = array('title'=>'info', 'content'=>$this->N->description);
		}
		if(count($this->N->tag->all)>0)
		{
			$this->data['content']['tags'] = array('title'=>'tags', 'content'=>$this->load->view('gift/includes/tags', array('tags'=>$this->N->tag->all), true));
		}
		
		// Prep Buttons
		$button_data = array('page'=>"gift_dashboard","G"=>$this->N);
		$this->data['buttons'] = $this->load->view('gift/includes/buttons/need_dashboard', $button_data, true);
		
		$this->data['title'] = "Edit Gift";
		$this->data['N'] = $this->N;
		$this->load->view('header', $this->data);
		$this->load->view('gift/view', $this->data);
		$this->load->view("footer", $this->data);	
	}
	
	
}
