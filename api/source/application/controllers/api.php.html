application/controllers/api.php


class Api extends Controller {

	var $data;
	
	function __construct()
	{
		parent::Controller();
		$this->util->config();
		$this->data = $this->util->parse_globals();
	}

	public function member( $id, $method, $param = null )
	{
		if($method=="photo")
			$this->_member_photo( $id, $param );
		
	}
	protected function _member_photo( $id, $param = null )
	{
		$U = new User( $id );
		if(!$U->exists())
		{
			echo "User Does Not Exist.";
			return false;
		}
		if($U->facebook_photo==1&& !empty($U->facebook_id))
		{
			if($param=='thumb') $url = "http://graph.facebook.com/".$U->facebook_id."/picture?type=square";
			else $url = "http://graph.facebook.com/".$U->facebook_id."/picture?type=large";
		}
		else
		{
			$U->default_photo->get();
		 	if($U->default_photo->exists())
		 	{
		 		if($param == "thumb") $url = $U->default_photo->thumb_url;
		 		else $url = $U->default_photo->url;
		 		
		 		// Check to see if local file exists
		 		
		 		// first, generate path
		 		$path = $this->config->item('base_path');
		 		if($param == "thumb" ) $path  .= "uploads/thumbnails/".basename($url);
		 		else $path .= "uploads/".basename($url);
		 		
		 		// If it doesn't, use default URL
		 		if( !file_exists( $path ) )
		 		{
		 			$url =  base_url()."assets/images/default_user_photo.jpg";
		 		}
		 	}
		 	else
		 	{
		 		$url =  base_url()."assets/images/default_user_photo.jpg";
		 	}
		}

		header( "Location: ".$url);
	
	}
	function test()
	{
		$this->load->model('good');
		print_r($this->good->find());
	}

}
