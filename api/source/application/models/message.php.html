	function  compose()
	{
		/*
		* @Parameters
		*
		* (string) $subject
		* (string) $body 
		* (int) $author_id
		* (array) $recipients: array(recipient_id_1, recipient_id_2, recipient_id_3);
		* (int) $thread
		*/
		$this->validate();
		
		if ( empty($this->thread_id) )
		{
			$t = new Thread();
			$t->subject = $this->subject;
			$t->save();
			foreach($this->recipients as $val)
			{
				$u = new User($val);
				$t->save_recipient($u);
			}
			$this->thread_id = $t->id;
		}
		else
		{
			$t = new Thread($this->thread_id);
			$t->recipients->get();
			
			$this->recipients = array();
			
			foreach($t->recipient->all as $obj)
			{
				print_r($obj);
				$this->recipients[] = $obj->id;
			}
		}
		
		if( $this->save() )
			echo $this->id;
		else
			return false;
		
		/*
		*  Deliver the mail to each recipient 
		*  (ie insert a new row into message_threading)
		*/

		foreach ( $this->recipients as $key=>$val )
		{
			$d = new Message_delivery();
			
			$d->thread_id = $this->thread_id;
			$d->recipient_id = $val;
			// Mark sent messages as preopened
			if ( $key === "author" )
				$d->opened = 1;
			// More efficient way to save message_id
			$d->message_id = $this->id;
			$d->save();
			// The appropriate (and less efficient) way to save message_id
			//$this->save_message_delivery($d);
		}

		return true;
	}
	function transaction_dialogue()
	{
		/*
		* @Parameters
		*
		* (string) $body 
		* (int) $author_id
		* (array) $recipients: array(recipient_id_1, recipient_id_2, recipient_id_3);
		*/
		$this->type = 'transaction';
		
		$this->validate();
		if ( empty($this->transaction_id) )
			return false;
		if( !$this->save() )
			return false;
		
		/*
		*  Deliver the mail to each recipient 
		*  (ie insert a new row into message_threading)
		*/

		foreach ( $this->recipients as $key=>$val )
		{
			$d = new Message_delivery();
			
			$d->thread_id = $this->thread_id;
			$d->recipient_id = $val;
			// Mark sent messages as preopened
			if ( $key === "author" )
				$d->opened = 1;
			// More efficient way to save message_id
			$d->message_id = $this->id;
			$d->save();
			// The appropriate (and less efficient) way to save message_id
			//$this->save_message_delivery($d);
		}

		return true;
	}	
	
	function _check_recipients ( $field )
	{
		if ( !empty($this->recipients) && empty($this->thread_id) )
		{
			if(empty($this->recipients['author']))
			{
				$this->recipients['author'] = $this->author_id;
				$key = array_search($this->author_id, $this->recipients);
				unset($this->recipients[$key]);
			}
			return true;
		}
		elseif ( !empty($this->thread_id) )
		{
			$this->recipients = array();
			$m = new Message();
			$m->where('thread_id', $this->thread_id)->get();
			if(empty($this->recipients['author']))
			{
				$this->recipients['author'] = $m->author_id;
				$key = array_search($m->author_id, $this->recipients);
				unset($this->recipients[$key]);

			}
			$d = new Message_delivery();
			$d->where('message_id', $m->id)->get();
			foreach ( $d->all as $obj )
			{
				$this->recipients[] = $obj->recipient_id;
			}
			if ( count($this->recipients) > 1 )
				return true;
		}
		return false;
	}
    
}
