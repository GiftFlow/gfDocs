application/libraries/notify.php

if (!defined('BASEPATH'))
      exit('No direct script access allowed');
/**
 *	The Notify library handles all notifications and alerts
 *
 *	@author Brandon Jackson
 * 	@package Libraries
 */

class Notify
{
      public $from_email;
      public $from_name;
      public $to;
      public $reply_to_email;
      public $reply_to_name;
      public $cc;
      public $bcc;
      public $subject;
      public $parseables;
      public $template;
      public $l_delimiter;
      public $r_delimiter;
      
      protected $CI;
      
      public function __construct($config = array())
      {
            $this->from_email = $config["from_email"];
            $this->from_name  = $config["from_name"];
            $this->to  = $config["to"];
            $this->reply_to_email = $config["reply_to_email"];
            $this->reply_to_name  = $config["reply_to_name"];
            $this->cc  = $config["cc"];
            $this->bcc  = $config["bcc"];
            $this->subject = $config["subject"];
            $this->parseables = $config["parseables"];
            $this->template   = $config["template"];
            $this->l_delimiter  = ($config["l_del"] == "") ? "{{" : $config["l_del"];
            $this->r_delimiter   = ($config["r_del"] == "") ? "}}" : $config["r_del"];
            
            $this->CI =& get_instance();
            $this->CI->load->library('util');
            $this->CI->load->library('email');
      }
      
      function send_email()
      {
            $this->CI->email->from($this->from_email, $this->from_name);
            $this->CI->email->to($this->to);
            
            if ($this->reply_to_email != "")
                  $this->CI->email->reply_to($this->reply_to_email, $this->reply_to_name);
            
            if ($this->cc != "")
                  $this->CI->email->cc($this->cc);
            
            if ($this->bcc != "")
                  $this->CI->email->bcc($this->bcc);
            
            $this->CI->email->subject($this->subject);
            $this->CI->email->message($this->template_parser());
            
            if (!$this->CI->email->send())
            {
                  return TRUE;
            }
            
            return FALSE;
      }
      
      function template_parser()
      {
            foreach ($this->parseables as $key => $val)
            {
                  $find_arr[]    = $this->l_delimiter . $key . $this->r_delimiter;
                  $replace_arr[] = htmlentities($val);
            }
            
            return str_replace($find_arr, $replace_arr, $this->template);
      }
}
