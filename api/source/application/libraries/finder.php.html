	public function __construct()
	{
		$this->CI =& get_instance();

		else

			$this->format = "object";

	/**
	*	Geocodes an IP address using an external IP address API.
	*	If the IP is encoded successfully, then it will populate an array and various
	*	properties of $this.
	*
	*	@param string $ip
	*	@return array
	*/
	public function locate_ip( $ip = NULL )
	{
		if(empty($ip)) $ip = $this->CI->input->ip_address();
		$url = "http://ipinfodb.com/ip_query.php?ip=".$ip."&timezone=false";
		$xml = simplexml_load_file( $url );
		if(!empty($xml->City)&&!empty($xml->RegionName))
		{
			$data['city'] = $xml->City;
			$data['state'] = $xml->RegionName;
			$data['latitude'] = $xml->Latitude;
			$data['longitude'] = $xml->Longitude;
			
			$this->address = $data['city'].', '.$data['state'];
			$this->latitude = $data['latitude'];
			$this->longitude = $data['longitude'];
			
			return $data;
		}
		else return false;
	}

	/**
	*	Searches gifts based on their tags
	*	@param string $tag	 Tag name to search by
	*	@return object/JSON
	*/
	public function tag_search ( $tag = NULL )
	{
		if(!empty($tag))
		{
			// Find the tag_id
			$tag_query = $this->CI->db->select("id")->like('name', $tag)->get('tags');
			if($tag_query->num_rows==0) return false;
			$tag = $tag_query->row();
			$this->tag_id = $tag->id;
		}
		$this->result = $this->CI->db->query( $this->_tag_query() );

		return $this->_render_output();
	}
	
	/**
	*	Generates results for the find page.
	*	It uses the search keyword ($this->keyword) to search for gifts based
	*	on location, tags and titles.
	*	@return JSON
	*/
	public function geosearch()
	{
		// Excludes the logged in user from current search
		$this->_exclude_logged_in_user();
	
		// Searches good titles and tags for the provided keyword
		$this->keyword_search();
		
		// Formats result into a datamapper-like object
		$this->_object_factory();

		// Render output as either JSON or an object
		return $this->_render_output();
	}
	
	public function giftflow()
	{
		if(!empty($_REQUEST['source']) ) $source = $_REQUEST['source'];
		else $source = 'following';
				
		if( $source == "following" )
		{
			$this->follower_search();
		}
		elseif( $source == "nearby_users" )
		{
			$this->nearby_users();
		}
		elseif( $source == "nearby_gifts" )
		{
			$this->nearby_gifts();
		}
		
		return $this->_render_output();
	}
	
	/**
	*	Follower search
	*/
	public function follower_search()
	{
		$user_id = $this->CI->session->userdata['user_id'];
		$q =$this->CI->db	->select('U.id AS user_id')
							->from("users AS U ")
							->join("followings_users AS FU ", "FU.following_id=U.id")
							->where("FU.user_id", $user_id)
							->get();
		foreach( $q->result() as $key=>$val )
		{
			$this->users[] = $val->user_id;
		}
		$this->CI->db->where('U.id !=', $user_id); 
		return $this->user_search();
	}
	
	/**
	*	Performs generic user search based on user_id(s) provided.
	*/
	function user_search()
	{
		if( is_array( $this->users ) )
		{
			$this->CI->db->where_in('U.id', $this->users);
		}
		elseif( is_numeric( $this->users ) )
		{
			$this->CI->db->where('U.id', $this->users);
		}
		$this->_basic_query();
		$this->result = $this->CI->db->get()->result();
		$this->_object_factory();
		return $this->object;
	}
	
	/**
	*	Searches based on tag and title similarity
	*
	*	$this->search_keyword is required
	*/
	public function keyword_search()
	{
		// Search by keyword
		$this->subquery[] = $this->_title_query();

		// If keyword matches a tag, filter by it
		$tag_query = $this->_tag_query();
		if( !empty($tag_query) )
			 $this->subquery[] = $tag_query;
		
		// Unite subqueries
		$this->_subquery_union();
			
		// Store result
		$this->result = $this->query->result();
	}

	public function where( $key, $val = NULL )
	{
		$this->CI->db->_reset_select();
		if( is_array( $key ) )
			$this->CI->db->where($key);
		elseif( !empty($val))
			$this->CI->db->where($key, $val);
		else
			return false;
		$this->where[] = substr($this->CI->db->_compile_select(),15);
		$this->CI->db->_reset_select();

		return $this;
	}
	
	/**
	*	Basic query that is run every time for both subqueries and normal queries.
	*	There are several subroutines which are run that add to the active record
	*	object's methods.
	*/
	protected function _basic_query()
	{
		
		// Determine geographic boundaries
		$this->_set_geographic_bounds();
		
		// Find goods
		$this->CI->db	->select("G.id AS good_id")
						->select("G.type AS good_type")
						->select("G.title AS good_title")
						->select("G.description AS good_description")
						->select("G.created AS good_created")
						->select("L.id AS location_id")
						->select("L.title AS location_title") // useful?
						->select("L.address AS location_address")
						->select("L.latitude AS location_latitude")
						->select("L.longitude AS location_longitude")
						->select("L.street_address AS location_street_address")
						->select("L.city AS location_city")
						->select("L.state AS location_state")
						->select("L.zip AS location_zip")
						->select("P.id AS photo_id")
						->select("P.url AS photo_url")
						->select("P.thumb_url AS photo_thumb_url")
						->select("P.caption AS photo_caption") // useful?
						->select("U.id AS user_id")
						->select("U.username AS user_username") // used?
						->select("U.email AS user_email")
						->select("U.type AS user_type")
						->select("U.first_name AS user_first_name")
						->select("U.last_name AS user_last_name")
						->select("U.screen_name AS user_screen_name")
						->select("U.bio AS user_bio")
						->select("U.created AS user_created")
						->select("U.occupation AS user_occupation")
						->select("U.phone AS user_phone") // add privacy check
						->select("U.facebook_id AS user_facebook_id")
						->select("UP.id AS user_photo_id")
						->select("UP.url AS user_photo_url")
						->select("UP.thumb_url AS user_photo_thumb_url")
						->select("UP.caption AS user_photo_caption")
						->select("C.id AS catgory_id")
						->select("C.name AS category_name")
						->from("goods AS G ")
						->join("users AS U", "G.user_id = U.id")
						->join("locations AS L ", "L.id = G.location_id")
						->join("photos AS P ", "G.default_photo_id = P.id", "left")
						->join("photos AS UP ", "U.default_photo_id = UP.id", "left")
						->join("categories AS C", "C.id = G.category_id", "left")
						->where('G.type', $this->type);
		
		// Iterate through defined where conditions
		if( !empty( $this->where ) )
		{
			foreach( $this->where as $val )
			{
				$this->CI->db->where($val);
			}
		}
		
		// Count related transactions
		$this->_count_transactions();
		
		// Sort & Paginate
		$this->_sort_and_paginate();
	}
	
	/**
	*	Counts the requests for each good, grouped by request status
	*/
	protected function _count_transactions()
    	{
    		$this->CI->db	->select("COUNT(T_pending.id) AS good_pending_requests")
						->select("COUNT(T_ignored.id) AS good_ignored_requests")
						->select("COUNT(T_accepted.id) AS good_accepted_requests")
						->select("COUNT(T_completed.id) AS good_completed_requests")
						->join("goods_transactions AS GR ", "GR.gift_id = G.id", "left")
						->join("transactions AS T_pending ", " T_pending.id = GR.request_id AND T_pending.status='pending'", "left")
						->join("transactions AS T_ignored ", " T_ignored.id = GR.request_id AND T_ignored.status = 'ignored'", "left")
						->join("transactions AS T_accepted ", " T_accepted.id = GR.request_id AND T_accepted.status = 'accepted'", "left")
						->join("transactions AS T_completed ", " T_completed.id = GR.request_id AND T_completed.status = 'completed'", "left")
						->group_by("G.id");
    	}
    	
    	/**
    	*	If the user is logged in, this function fetches their default location and
    	*	sends its information into Finder's location-related properties.
    	*	@return boolean
    	*/
	protected function _set_logged_in_user_location()
	{
		if( !empty( $this->CI->session->userdata['user_id'] ))
		{
			$this->U = new User($this->CI->session->userdata['user_id']);
			$this->U->default_location->get();
			if( $this->U->default_location->exists() )
			{
				$this->latitude = $this->U->default_location->latitude;
				$this->longitude = $this->U->default_location->longitude;
				$this->address = $this->U->default_location->address;
				return true;
			}
			else
			{
				$ip = $this->locate_ip();
				if( !empty( $ip ) ) 
					return true;
			}
		}
		return false;
	}


	/**
	*	Adds where clauses that limit results to a specific geographic area
	*/
	protected function _set_geographic_bounds()
	{
		// Geocode if needed
		if(!empty($this->location_keyword)&&(empty($this->latitude)||empty($this->longitude)))
		{
			$this->_geocode();
		}
		
		// Make sure latitude and longitude are present
		if( empty($this->latitude) || empty($this->longitude) )
		{
			return false;
		}
		
		// If bounds aren't set, find and set them
		if( empty($this->bounds) )
		{
			// How wide is one degree?
			$latitude_degree = 69;
			$longitude_degree = ( ( 3.14 * cos(deg2rad($this->latitude)) * 6367449 ) / (180 * 1000 ) ) / 1.609;
			
			// How wide is the search radius (in degrees)?
			$latitude_radius = $this->radius  / $latitude_degree;
			$longitude_radius = $this->radius / $latitude_degree;
			$this->bounds = array ( "latitude"=>array ( 	"upper"=> ( $this->latitude + $latitude_radius ),
											"lower" => ( $this->latitude - $latitude_radius ) ),
							"longitude"=>array(	"upper"=> ( $this->longitude + $longitude_radius ),
											"lower" => ( $this->longitude - $longitude_radius ) ) );
		}
		if( !empty( $this->bounds ) )
		{
			// Add clauses to query string
			$this->CI->db->where("L.latitude BETWEEN ".$this->bounds['latitude']['lower']." AND ".$this->bounds['latitude']['upper']);
			$this->CI->db->where("L.longitude BETWEEN ".$this->bounds['longitude']['lower']." AND ".$this->bounds['longitude']['upper']);
			$this->CI->db->select("( 3959 * acos( cos( radians( ".$this->latitude." ) ) * cos( radians( L.latitude ) ) * cos( radians( L.longitude ) - radians(".$this->longitude.") ) + sin( radians(".$this->latitude.") ) * sin( radians( L.latitude ) ) ) ) AS location_distance");
		}
	}
	
	/**
	*	Sorts and paginates search results
	*	First, it checks to see if the user has input any order_by or limit information
	*	using the Finder::order_by() or Finder::limit() methods.
	*	If not, then reverts to default values specified in the model's properties
	*/
	protected function _sort_and_paginate()
	{
		if(!empty($this->order_by_field))
			$this->CI->db->order_by($this->order_by_field, $this->order_by_direction);
		else
			$this->CI->db->order_by($this->default_order_by_field, $this->default_order_by_direction);
		
		if(!empty($this->limit_size))
			$this->CI->db->limit($this->limit_size, $this->limit_offset);
		else
			$this->CI->db->limit($this->default_limit_size, $this->default_limit_offset);
	}
	
	
	/*
	*	Subqueries
	*/
	
	/**
	*	Searches by tag
	*
	*	@return string
	*/
	protected function _tag_query()
	{
		// Only proceed if tag id is already set or if a search finds a match
		if( !empty( $this->tag_id ) ||  $this->_set_tag( $this->search_keyword ) )
		{
			$this->CI->db->_reset_select();
			$this->_basic_query();
			$this->CI->db->join("goods_tags AS GT ", "GT.good_id=G.id")
				->join("tags AS T ", "GT.tag_id=T.id")
				->where('T.id = '.$this->tag_id);
			$subquery = $this->CI->db->_compile_select();
			$this->CI->db->_reset_select();
			return $subquery;
		}
		else
		{
			return false;
		}
	}
	
	/**
	*	Searches by title
	*	@return string
	*/
	protected function _title_query()
	{
		$this->CI->db->_reset_select();
		$this->_basic_query();
		$this->CI->db->like("G.title", $this->search_keyword);
		$subquery = $this->CI->db->_compile_select();
		$this->CI->db->_reset_select();
		return $subquery;
	}
	
	/**
	*	Performs a union query if there are multiple subqueries.
	*/
	protected function _subquery_union()
	{
		if(count($this->subquery)==2)
		{
			// In order to get the proper order_by and limit clauses,
			// we have to perform some trickery.
			
			// $this->_sort_and_paginate() sets the database active record
			// object with the proper clauses, but it has SELECT * at the
			// beginning which we don't need. So we substring that part out. Done.
			$this->CI->db->_reset_select();
			$this->_sort_and_paginate();
			$sorting = substr($this->CI->db->_compile_select(),9);
			$this->CI->db->_reset_select();
			$union = "(".$this->subquery[0]." ) UNION ( ".$this->subquery[1].")". $sorting;
			$this->query = $this->CI->db->query($union);
		}
		elseif(count($this->subquery)==1)
		{
			$this->query = $this->CI->db->query($this->subquery[0]);
		}
		if(!empty($this->query)) return $this->query;
	}

	/**
	*	Exclude logged in user
	*/
	protected function _exclude_logged_in_user()
	{
		if( !empty( $this->CI->session->userdata['user_id'] ) )
		{
			$this->where('U.id != ', $this->CI->session->userdata['user_id']);
		}
	}

	
	/**
	*	Turns a DB result object into datamapper object clone
	*/
	protected function _object_factory()
	{
		foreach( $this->result as $row )
		{
			$item = (object) array(
									"user"=>new stdClass(),
									"location"=>new stdClass(),
									"default_photo"=>new stdClass(),
									"category"=>new stdClass()
									);
			foreach( $row as $key=>$val )
			{
				$object = substr($key, 0, strpos($key, "_"));
				$key = substr(strstr($key, "_"), 1);
				if( $object == "good")
					$item->$key = $val;
				else
					$item->$object->$key = $val;
			}
			$this->object[] = $item;
		}
	}
	
	/**
	*	Geocodes this->address into a usable location object
	*/
	protected function _geocode()
	{
		$L = new Location();
		if(!$L->geocode( $this->location_keyword )) return false;
		$this->location = $L;
		$this->latitude = $L->latitude;
		$this->longitude = $L->longitude;
		$this->address = $L->address;
	}
	
	/**
	*	Formats results into HTML for display on find page
	*/
	protected function _format_result()
	{
		if(!empty($this->object))
		{
			foreach($this->object as $key=>$val)
			{
				$val->html = result_row($val, array("author", "location"));
			}
		}
		else
			return false;
	}
	
	/**
	*	Takes the keyword input and tries to find a matching tag
	*
	*	@param string $tag 	Name of tag to search for
	*/
	protected function _set_tag( $tag )
	{
		// Find the tag_id
		$tag_query = $this->CI->db->select("id")->like('name', $tag)->get('tags');
		if($tag_query->num_rows==0) return false;
		$tag = $tag_query->row();
		// Set tag_id
		$this->tag_id = $tag->id;
		return true;
	}
	protected function _render_output()
	{
		if( $this->format == "JSON" )
		{
			$this->_format_result();
			if( empty($this->latitude) || empty($this->longitude))
				$this->_set_logged_in_user_location();
			if(!empty($this->latitude)&&!empty($this->longitude)&&!empty($this->address))
				$data['center'] = (object) array ( "latitude"=>$this->latitude, "longitude"=> $this->longitude, "address"=>$this->address);
			$data['total_results'] = count($this->object);
			$data['results'] = $this->object;
			return json_encode( $data );
		}
		else
		{
			$this->_format_result();
			return $this->object;
		}
	}

}
