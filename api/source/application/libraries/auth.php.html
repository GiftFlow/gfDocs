/**
*	The Auth library handles both authentication and authorization functionality.
*
*	@author Brandon Jackson
* 	@package Libraries
*/
class Auth
{

	protected $CI;
	
	/**
	*	@var User
	*/
	var $U;

	/**
	*	Constructor
	*/
	public function __construct()
	{
		$this->CI =& get_instance();
	}
	
	/**
	*	Register new user
	*
	*	Reads from $_POST data
	*
	*	@return User
	*/
	public function register()
	{
		// Construct new User object
		$this->U = new User();
		
		// Set $_POST data
		$this->U->email = $this->CI->input->post('email');
		$this->U->screen_name = $this->U->email;
		$this->U->password = $this->CI->input->post('password');
		
		// Set default user role to 2, which is a normal user
		$this->U->role_id = 2;
		
		// Set IP address
		$this->U->ip_address = $this->CI->input->ip_address();
		
		// Generate salt based on time and email
		$this->U->salt =  sha1('~'.$this->U->email.'~'.microtime(TRUE));
		
		// Save new user. If successful....
		if($this->U->save())
		{ 
			// Create a new settings object.
			$s = new User_setting();
			
			// Save settings object
			$this->U->save($s);
			
			// Send email verification
			$this->email_verification();
		}
		
		// Return new user
		return $this->U;
	}
	
	/**
	*	Handles logins. Reads credentials from $_POST and validates using 
	*	methods stored in the User model. If successful, creates new session.
	*
	*	@return User
	*/
	public function login()
	{
		// Construct new User object
		$this->U = new User();
		
		// Parse $_POST input
		$this->U->email = $this->CI->input->post('email');
		$this->U->password = $this->CI->input->post('password');
		
		// Validates login info. If valid...
		if($this->U->login())
		{
			// ... create new session
			$this->new_session();
		}
		
		// Return User object
		return $this->U;
	}
	/**
	*	Logs user out. Destroys CodeIgniter session.
	*
	*	@return void
	*/
	public function logout()
	{
		$this->CI->session->sess_destroy();
	}
	
	/**
	*	Creates new CodeIgniter session
	*
	*	@return boolean
	*/
	protected function new_session()
	{
		if ( empty($this->U) )
		{
			return FALSE;
		}
		$this->U->default_location->get();
		$this->U->default_photo->get();
		$this->U->role->get();
		$userdata = array ( 	'logged_in' => TRUE,
						'role_id'=>$this->U->role->id,
						'level'=>$this->U->role->level,
						'user_id'=>$this->U->id,
						'email'=>$this->U->email,
						'username'=>$this->U->username,
						'screen_name'=>$this->U->screen_name,
						'first_name'=>$this->U->first_name,
						'last_name'=>$this->U->last_name,
						'location_latitude'=>$this->U->default_location->latitude,
						'location_longitude'=>$this->U->default_location->longitude,
						'location_street_address'=>$this->U->default_location->street_address,
						'location_address'=>$this->U->default_location->address,
						'location_city'=>$this->U->default_location->city,
						'location_state'=>$this->U->default_location->state,
						'status'=>$this->U->status
						);
		if($this->U->facebook_photo==1&&!empty($this->U->facebook_id))
		{
			$userdata['photo_thumb_url'] = "http://graph.facebook.com/".$this->U->facebook_id."/picture?type=square";
			$userdata['photo_url'] = "http://graph.facebook.com/".$this->U->facebook_id."/picture?type=large";
		}
		elseif(!empty($this->U->default_photo->thumb_url)&&!empty($this->U->default_photo->url))
		{
			$userdata['photo_thumb_url'] = $this->U->default_photo->thumb_url;
			$userdata['photo_url'] = $this->U->default_photo->url;
		}
		else
		{
			$userdata['photo_thumb_url'] = base_url().'assets/images/default_user_photo_thumb.jpg';
			$userdata['photo_url'] = base_url().'assets/images/default_user_photo.jpg';
		}
		// Save to userdata array
		$this->CI->session->set_userdata($userdata);
		return TRUE;
	}
	
	
	/*
	*
	*	AUTHORIZATION FUNCTIONS
	*
	*/

	/**
	*	Validates whether the user has a high enough authentication level to view a page.
	*	
	*	@param int $min	The minimum authorization level needed to view this page
	*
	*	@return boolean
	*/
	public function validate( $min )
	{
		if($this->CI->session->userdata('level')) $level = $this->CI->session->userdata('level');
		else $level = 0;

		if(empty($min))
		{
			$method = $this->CI->uri->segment(2);
			if(empty($method)) $method = 'index';
			$where = array ('controller' => $this->CI->uri->segment(1), 'method' => $method );
			$query = $this->CI->db->where($where)->get('permissions');
			if($query->num_rows() == 1)
			{
				$row = $query->row();
				$min = $row->level;
			}
			else $min = 100;
		}
		if ( $level >= $min )
			return true;
		else
			return false;
	}
	
	/**
	*	Lets authorized users view a page and redirects unauthorized users
	*	to either an access denied page or login page.
	*
	*	@param int $min 			Minimum auth level
	*	@param string $redirect	URL to redirect to
	*/
	
	public function bouncer ($min = NULL, $redirect = 'login')
	{
		if($this->validate($min))
		{
			return true;
		}
		else
		{
			if($this->CI->session->userdata('logged_in'))
			{
				redirect('restricted');
			}
			else
			{
				$this->CI->db->insert('redirects', array("url"=>$this->CI->uri->uri_string()));
				redirect($redirect."/".$this->CI->db->insert_id());
			}
		}
	}
	
	
	function email_verification()
	{
		$this->U->deactivate();
		$this->CI->load->library('email');
		
		$config['protocol']  = 'smtp';
		$config['newline'] = '\r\n';
		$config['crlf'] = '\r\n';
		$config['smtp_host'] = 'mail.giftflow.org';
		$config['smtp_user'] = 'no-reply@giftflow.org';
		$config['smtp_pass'] = 'mon3yis3vil';
		$config['smtp_port'] = '26';
		$config['mailtype']  = 'text';
		$config['charset']   = 'utf-8';

		$this->CI->email->initialize($config);
		

		$this->CI->email->from('no-reply@giftflow.org', 'GiftFlow');
		$this->CI->email->to($this->U->email);
		
		$this->CI->email->subject('[GiftFlow] Activate Your Account');
		$message = "Hello! \r\n";
		$message .= "Welcome to GiftFlow. Please click the link below to activate your account and verify this email address. \r\n ";
		$message .= site_url('member/activate/'.$this->U->activation_code)." \r\n";
		$message .= "Thanks! \r\n";
		$message .= "The GiftFlow Team";
		
		$this->CI->email->message($message);
		
		$this->CI->email->send();		
	}
	
	function facebook( $data )
	{
		/** 	Outline
		* 	1: Is the Facebook data valid?
		*    		a: Yes. Proceed to 2.
		*    		b: No. Return error message.
		* 	2: Is Facebook ID in system?
		*     	a: Yes. Create session for that user.
		*     	b: No. Facebook ID not in system. Proceed to 3.
		*	3. Is this user already logged in?
		*		a. Yes. Link Facebook ID with that user. Fill in missing profile fields.
		*		b. No. Proceed to step 4.
		* 	4: Is email in system?
		*     	a: Yes. Create session for that user. Link Facebook ID with that user. Fill in missing profile fields.
		*     	b: No. Create new User.
		*/
		if(!empty($data->id))
		{
			$step_two = $this->CI->db->where('facebook_id',$data->id)->from('users')->get();
			
			// Outcome 2a
			if($step_two->num_rows==1)
			{
				$user = $step_two->row();
				$this->U = new User($user->id);
				$this->U->facebook_sync( $data );
				$this->new_session();
				redirect('');
			}
			// eof 2a
			
			elseif(!empty($this->CI->session->userdata['user_id']))
			{
				// Outcome 3a
				$this->U = new User( $this->CI->session->userdata['user_id'] );
				$this->U->facebook_sync($data);
				$this->CI->session->set_flashdata('success', 'Facebook account now linked with GiftFlow');
				redirect('account/links');
				// eof 3a
				
			}
			elseif(!empty($data->email))
			{
				$step_four = $this->CI->db->where('email',$data->email)->from('users')->get();
				// Outcome 4a
				if($step_four->num_rows==1)
				{
					$user = $step_four->row();
					$this->U = new User($user->id);
					$this->U->facebook_sync($data);
					
					// Create new session
					if($this->new_session())
					{
						$this->U->save();
						// Redirect to dashboard
						redirect('');
					}
				}
				// eof 4a
				
				// Outcome 4b
				// New user
				else
				{
					$this->U = new User();
					if($this->U->facebook_sync( $data ))
					{
						if($this->new_session())
						{
							// Save new user
							$this->U->save();
							
							// Redirect to facebook welcome page
							redirect('member/facebook_welcome');
						}
					}
				}
				// eof 4b
			}
		}
		$this->U = new User();
		if(!empty($data->id))
			$this->U->facebook_id = $data->id;
		if(!empty($data->name))
			$this->U->screen_name = $data->name;
		if(!empty($data->first_name))
			$this->U->first_name = $data->first_name;
		if(!empty($data->last_name))
			$this->U->last_name = $data->last_name;
		if(!empty($data->email))
			$this->U->email = $data->email;
		
	}
	
	/**
	*	Lists pages and looks for their permissions. DEPRECATED.
	*
	*	@deprecated
	*/
	public function page_lister()
	{
		$this->invalid_methods = array ("Controller","_ci_initialize","_ci_scaffolding", "CI_Base", "get_instance", "__construct");

		$this->APPLICATION_DIRECTORY = "/Applications/XAMPP/xamppfiles/htdocs/gift/system/application/controllers/";
		
		// Open the Application Directory
		$directory = opendir($this->APPLICATION_DIRECTORY);

		// loop through all the files and set the name into an array
		while (($file = readdir($directory)) !== FALSE)
		{
			if (preg_match('/.php/i',$file))
			{
				$temp = explode('.',$file);
				$controllers[] = $temp[0];
			}
		}

		// Close the Application Directory
		closedir($directory);

		// Look through all the controllers for their methods
		foreach ($controllers as $c)
		{
			// Check to see if the class has been loaded
			if (!class_exists($c))
			{
				$this->CI->load->file($this->APPLICATION_DIRECTORY.$c.'.php');
			}

			// Add the constructors to the values we don't care about
			$this->invalid_methods[] = ucfirst($c);

			foreach (get_class_methods($c) as $key => $val)
			{
				  /* Get a reflection object for the class method */
       				$reflect = new ReflectionMethod($c, $val);

				/* For private, use isPrivate().  For protected, use isProtected() */
				
				if($reflect->isPublic()) 
				{
				  	if (!in_array($val,$this->invalid_methods))
					{
						// Put the methods we care about into an array
						$method[$c][] = $val;
					}
				}
			}
		}
		
		// Now lets create the classes and methods and relate them
		foreach ($method as $class => $methods)
		{
			foreach ($methods as $value)
			{
				$where = array('controller'=>$class, 'method'=>$value);
				$query = $this->CI->db->where($where)->get('permissions');
				if($query->num_rows == 0)
				{
					$this->CI->db->insert('permissions', $where);
				}
			}	
		}
	}
