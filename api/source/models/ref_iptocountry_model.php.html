class Ref_iptocountry_model extends Model {
	
	/**
	 * Constructor
	 */
	function Ref_iptocountry_model()
	{
		### Call the Model constructor
		parent::Model();
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Get country by ip address
	 *
	 * @access public
	 * @param string $ip
	 * @return string
	 */
	function get_by_ip($ip)
	{
		// Unsigned long representation of ip address
		$ulip = sprintf("%u", ip2long($ip));
		
		$query = $this->db->get_where('ref_iptocountry', array('ip_from <' => $ulip, 'ip_to >' => $ulip));
		
		if ($row = $query->row()) {
			return $row->country_code;
		}
		return FALSE;
	}
	
}
