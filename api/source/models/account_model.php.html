class Account_model extends Model {
	
	/**
	 * Constructor
	 */
	function Account_model()
	{
		// Call the Model constructor
		parent::Model();
	}
	
	// --------------------------------------------------------------------
	
	/**
	 * Get account by id
	 *
	 * @access public
	 * @param string $account_id
	 * @return object account object
	 */
	function get_by_id($account_id)
	{
		return $this->db->get_where('users', array('id' => $account_id))->row();
	}
	
	/**
	 * Get account by username
	 *
	 * @access public
	 * @param string $username
	 * @return object account object
	 */
	function get_by_username($username)
	{
		return $this->db->get_where('users', array('email' => $username))->row();
	}
	
	
	// --------------------------------------------------------------------
	
	/**
	 * Update account details
	 *
	 * @access public
	 * @param int $account_id
	 * @param array $attributes
	 * @return void
	 */
	function googleupdate($account_id, $attributes)
	{
		// Update
		if ($this->get_by_account_id($account_id))
		{
			$this->db->where('account_id', $account_id);
			$this->db->update('google_account_details', $attributes);
		}
		// Insert
		else
		{
			$attributes['account_id'] = $account_id;
			$this->db->insert('google_account_details', $attributes);
		}
	}
	
	
	function get_by_account_id($account_id)
	{
		return $this->db->get_where('google_account_details', array('account_id' => $account_id))->row();
	}
	// --------------------------------------------------------------------
	
	/**
	 * Create an account
	 *
	 * @access public
	 * @param string $username
	 * @param string $hashed_password
	 * @return int insert id
	 */
	function create($username, $email = NULL, $hashed_password = NULL)
	{
		$this->load->helper('date');
		
		$this->db->insert('users', array(
			'username' => $username, 
			'role_id' =>2,
			'email' => $email, 
			'password' => $hashed_password, 
			'created ' => mdate('%Y-%m-%d %H:%i:%s', now())
		));
		
		return $this->db->insert_id();
	}
	
	
}
